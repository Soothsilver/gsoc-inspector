55  /*

57  *
58  * @param stackDepth How many stack frames (from the current call stack) should be exited before the breakpoint hit (on the first such instruction). To leave
59  * just the current methods set to 1.

61  public ExpressionBreakpointStepOut (ThreadInfo ti, int stackDepth) {

74  public static ExpressionBreakpointStepOut getStepOutToCaller (JPFInspector inspector, int threadId, int stackDepth) throws JPFInspectorGenericErrorException {
75  assert inspector != null;
76  StopHolder sh = inspector.getStopHolder();
77  assert sh != null;
78  InspectorState inspState = sh.getInspectorState();
79  if (inspState == null) {
80  throw new JPFInspectorGenericErrorException(
81  "cannot create given breakpoint, JPF has to be connected and stopped - program state is required to create given breakpoint.");
82  }
83 
84  VM vm = inspState.getJVM();
85  assert (vm != null);
86 
87  // Find thread with given ThreadID
88  KernelState ks = vm.getKernelState();
89  ThreadList tl = ks.getThreadList();
90 
91  ThreadInfo ti = null;
92 
93  Iterator<ThreadInfo> it = tl.iterator();
94  while (it.hasNext()) {
95  ThreadInfo itTi = it.next();
96  assert (itTi != null);
97 
98  if (itTi.getId() == threadId) {
99  ti = itTi;
100  break;
101  }
102  }
103 
104  if (ti == null) {
105  throw new JPFInspectorGenericErrorException("Thread with index " + threadId + " not exists");
106  }
107 
108  return new ExpressionBreakpointStepOut(ti, stackDepth);
109  }
