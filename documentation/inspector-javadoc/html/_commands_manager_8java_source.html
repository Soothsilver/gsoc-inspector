180  @Override
181  public void forwardStep (StepType type) throws JPFInspectorGenericErrorException {
182  Boolean wasStopped = initialStopTest(true, "cannot execute forward step");
183  if (wasStopped == null) {
184  return;
185  }
186 
187  // TODO Create inspector.server class copy ... don't use inspector.client package version !!
188  CmdBreakpointCreate.ConsoleBreakpointCreationExpression newBP = new CmdBreakpointCreate.ConsoleBreakpointCreationExpression();
189  newBP.setBounds(null, null, "<=", 1);
190  newBP.setState(BreakPointStates.BP_STATE_ENABLED);
191 
192  ExpressionBoolean bpExpression = null;
193  if (type == StepType.ST_TRANSITION_DATA) {
194  bpExpression = new ExpressionBreakpointChoiceGenerator(BreakPointModes.BP_MODE_CHOICE_DATA);
195  } else if (type == StepType.ST_TRANSITION_SCHED) {
196  bpExpression = new ExpressionBreakpointChoiceGenerator(BreakPointModes.BP_MODE_CHOICE_SCHEDULING);
197  } else if (type == StepType.ST_TRANSITION_ALL) {
198  bpExpression = new ExpressionBreakpointChoiceGenerator(BreakPointModes.BP_MODE_CHOICE_BOTH);
199  } else if (type == StepType.ST_INSTRUCTION) {
200  bpExpression = new ExpressionBreakpointInstructionType(InstructionTypes.IT_ANY);
201  } else if (type == StepType.ST_LINE) {
202  bpExpression = new ExpressionBreakpointSingleStep(inspector, stopHolder.getJVM(), LocationTypes.LT_POSITION_LEAVED_STEP_OVER);
203  } else if (type == StepType.ST_STEP_IN) {
204  bpExpression = new ExpressionBreakpointSingleStep(inspector, stopHolder.getJVM(), LocationTypes.LT_POSITION_LEAVED_STEP_IN);
205  } else if (type == StepType.ST_STEP_OUT) {
206  ThreadInfo ti = MigrationUtilities.getLastThreadInfo(stopHolder.getJVM());
207  bpExpression = ExpressionBreakpointStepOut.getStepOutToCaller(ti);
208  } else {
209  throw new RuntimeException("Unsupported " + type.getClass().getSimpleName() + " entry " + type);
210  }
211 
212  assert (bpExpression != null);
213  breakpointMgr.createInternalBreakpoint(newBP, bpExpression, true);
214 
215  stopHolder.resumeExecution();
216  }

242  public boolean initialStopTest (boolean wait, String msg) throws JPFInspectorGenericErrorException {
243  if (inspector.getJPF() == null) {
244  throw new JPFInspectorGenericErrorException("No JPF instance to observe");
245  }
246  boolean wasStopped = stopHolder.isStopped();
247  if (!wasStopped) {
248  if (!wait) {
249  throw new JPFInspectorGenericErrorException("SuT is running - " + msg);
