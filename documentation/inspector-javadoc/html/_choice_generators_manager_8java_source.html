132  @Override
133  public void notifyChoiceGeneratorAdvance (ChoiceGenerator<?> cg, InspectorState inspState) {
134  if (DEBUG) {
135  out.println(this.getClass().getSimpleName() + ".notifyChoiceGeneratorAdvance( cg=" + cg + ", inspState=" + inspState + ")");
136  }
137 
138  // inspector.getDebugPrintStream().println("CG call occured - " + cg.toString() + " ... cg.className=" + cg.getClass().getName() + "#" + cg.hashCode());
139  CGTypes cgType = null;
140  if (cg instanceof ThreadChoiceGenerator) {
141  cgType = CGTypes.CG_TYPE_SCHEDULING;
142  } else {
143  cgType = CGTypes.CG_TYPE_DATA;
144  }
145 
146  // Following default trace if exists
147  int dftChoice = forwardTrace.getDefaultChoice(cg); // Default forward trace choice
148  if (dftChoice != ChoiceGeneratorsInterface.NO_DEFAULT_CHOICE) {
149  cg.reset();
150  cg.advance(dftChoice);
151  if (DEBUG) {
152  out.println(this.getClass().getSimpleName() + ".notifyChoiceGeneratorAdvance - using default choice ... dftChoice=" + dftChoice + ", cg=" + cg);
153  }
154  }
155 
156  // Check whether send notification to client
157  boolean printChoice = cgNotifications[getIndexCGNotificationSpecification(cgType, CGMode.CG_MODE_PRINT)].nofiticationEnabled();
158  boolean askChoice = cgNotifications[getIndexCGNotificationSpecification(cgType, CGMode.CG_MODE_ASK)].nofiticationEnabled();
159  if (printChoice == false && askChoice == false) {
160  return; // No notification are selected
161  }
162 
163  // Obtain all possible choices
164  int currentChoice = cg.getProcessedNumberOfChoices() - 1;
165  cg.reset();
166  int totalChoices = cg.getTotalNumberOfChoices();
167  String[] choices = new String[totalChoices];
168  for (int i = 0; i < totalChoices; i++) {
169  cg.advance();
170  Object genChoice = cg.getNextChoice();
171  choices[i] = (genChoice != null ? genChoice.toString() : "null");
172  }
173 
174  // Restore state of the CG
175  cg.reset();
176  for (int i = 0; i <= currentChoice; i++) {
177  cg.advance();
178  }
179 
180  int hashCode = cg.hashCode();
181 
182  // TODO Gather default choice if exists
183  callBacks.notifyChoiceGeneratorNewChoice(cgType, cg.getId(), hashCode, choices, currentChoice, ChoiceGeneratorsInterface.NO_DEFAULT_CHOICE);
184 
185  if (askChoice) {
186  callBacks.specifyChoiceToUse(totalChoices - 1);
187 
188  waitForChoice = true;
189  stopHolder.stopExecution(inspState);
190  waitForChoice = false;
191 
192  Object genChoice = cg.getNextChoice();
193  callBacks.notifyUsedChoice(cgType, cg.getId(), hashCode, cg.getProcessedNumberOfChoices(), (genChoice != null ? genChoice.toString() : "null"));
194  }
195  }

197  @Override
198  public void selectChoice (int selectedChoice) throws JPFInspectorGenericErrorException {
199  cmdMgr.initialStopTest(false, "'cg select' command can by called only if execution is stopped");
200  VM vm = stopHolder.getJVM();
201  assert vm != null;
202 
203  if (!waitForChoice) {
204  throw new JPFInspectorGenericErrorException("JPF has not been stopped by CG.Advanced call");
205  }
206 
207  ChoiceGenerator<?> cg = MigrationUtilities.getLastChoiceGenerator(vm);
208 
209  if (cg == null) {
210  return;
211  }
212  // Processing default forward trace
213  int dftChoice = forwardTrace.getDefaultChoice(cg); // Default forward trace choice
214  if (selectedChoice == ChoiceGeneratorsInterface.USE_DEFAULT_CHOICE) {
215  if (dftChoice == ChoiceGeneratorsInterface.NO_DEFAULT_CHOICE) {
216  throw new JPFInspectorGenericErrorException("no forward trace to follow");
217  }
218  selectedChoice = dftChoice;
219  }
220 
221  int totalChoices = cg.getTotalNumberOfChoices();
222 
223  // TODO Removed =
224  if (selectedChoice < 0 || selectedChoice >= totalChoices) {
225  // if (selectedChoice < 0 || selectedChoice > totalChoices) {
226  throw new JPFInspectorGenericErrorException("invalid index of the choices ( " + selectedChoice + ").");
227  } else {
228  if (selectedChoice != dftChoice) {
229  // Choice different path then default forward one -> destroy DFT is exists
230  forwardTrace.destroyTrace(true);
231  }
232  cg.reset();
233  selectedChoice++; // Own index is zero based however (ThreadInfo CG) needs to advance one to get first thread
234 
235  cg.advance(selectedChoice);
236  }
237 
238  stopHolder.resumeExecution();
239  }
