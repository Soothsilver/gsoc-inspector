89  @Override

91  initialStopTest(true, "threads");
92  Map<Integer, InstructionPosition> result = null;
93 
94  VM vm = getJVM();
95  if (vm == null) {
96  return null;
97  }
98 
99  ThreadList tl = vm.getThreadList();
100  if (threadNum != null) {
101  result = new HashMap<>(1);
102  } else {
103  result = new HashMap<>(tl.length());
104  }
105 
106  for (ThreadInfo ti : tl) {
107  if (!(threadNum == null || (ti.getId() == threadNum))) {
108  continue;
109  }
110  Instruction pc = ti.getPC();
111  if (pc == null) {
112  // Terminated threads -> no position to store
113  continue;
114  }
115  result.put(ti.getId(), InstructionPositionImpl.getInstructionPosition(pc));
116  }
117  return result;

122  initialStopTest(true, "expression cannot be evaluated");

124  // Create a parse tree
125  ExpressionStateRootNode<?> parsedExpr = parser.getExpressionStateInterface(expr);
126  if (parsedExpr == null) {
127  return null;

145  public boolean initialStopTest (boolean wait, String msg) throws JPFInspectorGenericErrorException {
146  if (inspector.getJPF() == null) {
147  throw new JPFInspectorGenericErrorException("No JPF instance to observe");
148  }
149  boolean wasStopped = stopHolder.isStopped();
150  if (!wasStopped) {
151  if (!wait) {
152  throw new JPFInspectorGenericErrorException("SuT is running - " + msg);
153  } else {
154  stopHolder.waitUntilStopped();
155  }
156  }
157  return wasStopped;
