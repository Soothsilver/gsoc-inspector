122  public WORKING_MODE setNewMode (WORKING_MODE newMode) {
123  synchronized (syncObj) {
124  WORKING_MODE oldMode = mode;
125  if (newMode != null) {
126  mode = newMode;
127 
128  // Unblock blocked callbacks (and/or Command thread)
129  if (newMode.equals(WORKING_MODE.WM_USER_COMMANDS)) {
130  syncObj.notifyAll();
131  }
132 
133  if (oldMode == WORKING_MODE.WM_USER_COMMANDS && newMode == WORKING_MODE.WM_EXECUTION_RECORD) {
134  // Start new replay of execution record -> clear errors
135  error = false;
136  errorMsg = null;
137  }
138  }
139  return oldMode;
140  }
141  }

147  // Block CB thread until Command thread specifies which CB is expected
148  // Null means pass any possible callback
149  protected void waitUntilCBIsSpecified (CB_METHODS calledCallback) {
150  synchronized (syncObj) {
151  if (mode == WORKING_MODE.WM_EXECUTION_RECORD) {
152 
153  while (method == null && mode == WORKING_MODE.WM_EXECUTION_RECORD) {
154  isCBwaiting = true;
155  try {
156  syncObj.wait();
157  } catch (InterruptedException e) {
158  // Ignore
159  }
160  isCBwaiting = false;
161  }
162 
163  if (method != null && mode == WORKING_MODE.WM_EXECUTION_RECORD) {
164  // Null means any callback is possible
165  if (!(CB_METHODS.CB_ANY.equals(calledCallback) || method.equals(calledCallback))) {
166  // Set error state
167  error = true;
168  errorMsg = "Unexpected callback. current callback: " + calledCallback + ", expected callback: " + method;
169  }
170  }
171  }
172  }
173  }
174 
175  // Blocks Command thread until first callback is executed
176  protected void waitForCB (CB_METHODS wait4method) {
177  synchronized (syncObj) {
178  if (mode == WORKING_MODE.WM_EXECUTION_RECORD) {
179 
180  while (method != null && mode == WORKING_MODE.WM_EXECUTION_RECORD) {
181  try {
182  syncObj.wait();
183  } catch (InterruptedException e) {
184  // Ignore
185  }
186  }
187 
188  if (mode == WORKING_MODE.WM_EXECUTION_RECORD) {
189  method = wait4method;
190 
191  if (isCBwaiting) {
192  // WakeUP waiting CB thread
193  syncObj.notifyAll();
194  }
195 
196  // Wait until CB processing is terminated
197  try {
198  syncObj.wait();
199  } catch (InterruptedException e) {
200  // Ignore
201  }
202  }
203  }
204  }
205  }
206 
207  // Called by CB thread to wake up blocked Command thread
208  protected void unblockCmdThread () {
209  synchronized (syncObj) {
210  assert (isCBwaiting == false);
211  method = null;
212  syncObj.notifyAll();
213  }
214  }

232 
233  if (cbStateChange_expectedState != null && !newState.equals(cbStateChange_expectedState)) {
234  error = true;

236  }
237  }

240 
241  unblockCmdThread();
242  }
243 
244  if (DEBUG) {
245  out.println(this.getClass().getSimpleName() + ".notifyStateChange( newState=" + newState + ", details=" + details + ") - end");
