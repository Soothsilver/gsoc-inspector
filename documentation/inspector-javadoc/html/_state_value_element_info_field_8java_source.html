136  public static StateReadableValueInterface createOuterClass (StateReadableValueInterface srvi) throws JPFInspectorException {
137  assert (srvi != null);
138 
139  ClassInfo ci = srvi.getClassInfo();
140  ClassInfo ciOuter = ci.getEnclosingClassInfo();
141  if (ciOuter == null) {
142  throw new JPFInspectorNotInnerClassException(ci);
143  }
144 
145  String newStateExpr = srvi.getStateExpr() + "." + PSEVariable.EXPRESSION_OUTER_CLASS;
146  ElementInfo ei = srvi.getReferenceValue();
147  if (StateValue.isStaticElementInfo(ei)) {
148  return StateElementInfo.createStaticOuterClass(srvi);
149  }
150  // Outer class can exists
151  if (ei == null) {
152  throw new JPFInspectorNotInnerClassException(ci);
153  }
154  // Implementation based on {@link DynamicElementInfo#getEnclosingElementInfo()}
155 
156  for (FieldInfo fi : ci.getDeclaredInstanceFields()) {
157  // Note name can be compiler dependent
158  // after the '$' is number representing depth of nesting
159  if (fi.getName().startsWith("this$")) {
160  return new StateValueElementInfoField(srvi, srvi.getReferenceDepth(), newStateExpr, ei, fi);
161  }
162  }
163 
164  assert (false); // Should be have outer class but not Field "this$"num not exists
165  throw new JPFInspectorNotInnerClassException(ci);
166  }
