57  public static char convertToChar (String str) throws JPFInspectorParsingErrorException {
58  assert (str != null);
59 
60  char result;
61  CharParsingState cps = new CharParsingState(str, '\'');
62 
63  // Check initial '
64  if (cps.strArr.length < 1) {
65  throw new JPFInspectorParsingErrorException("Empty character", str, 0);
66  }
67 
68  if (cps.strArr[0] != '\'') {
69  throw new JPFInspectorParsingErrorException("Error while parsing character at 0. The quote (\') character is expected in the beginning.", str, 0);
70  }
71  cps.pos++;
72 
73  result = parseCharRepresentation(cps);
74 
75  if (cps.pos >= cps.strArr.length) {
76  throw new JPFInspectorParsingErrorException("Error while parsing character at " + (str.length()) + ". The terminating quote (\') character is expected.",
77  str, str.length());
78  }
79 
80  if (cps.strArr[cps.pos] != '\'') {
81  throw new JPFInspectorParsingErrorException("Error while parsing character at " + (cps.pos) + ". The terminating quote (\') character is expected.", str,
82  cps.pos);
83  }
84 
85  if (cps.pos < cps.strArr.length - 1) {
86  throw new JPFInspectorParsingErrorException("Error while parsing character at " + (cps.pos + 1) + ". Garbage after terminating quote (\').", str,
87  cps.pos + 1);
88  }
89 
90  return result;
91  }

116  public static char parseCharRepresentation (CharParsingState cps) throws JPFInspectorParsingErrorException {
117  assert (cps != null);
118 
119  assert (cps.pos < cps.strArr.length);
120 
121  char c = cps.strArr[cps.pos];
122  cps.pos++;
123 
124  if (c != '\\') {
125  if (cps.bannedUnescapedCharacters.contains(c)) {
126  throw new JPFInspectorParsingErrorException("Error while parsing character at " + cps.pos + ". This character has to be escaped.", new String(
127  cps.strArr), cps.pos);
128  }
129  // Ordinary character
130  return c;
131  } else {
132  // Escaped chars \b \t \n \f \r \" \' \\
133  // or \377 -> octal code
134  // or \uFFFF
135 
136  if (cps.pos >= cps.strArr.length) {
137  // Invalid escape sequence ... \ char as last character in the input (no escaped character)
138  throw new JPFInspectorParsingErrorException("Error while parsing character at " + cps.pos
139  + ". No escape character specified. Escaped character expected.", new String(cps.strArr), cps.pos);
140  }
141  c = cps.strArr[cps.pos];
142  cps.pos++;
143 
144  if (c == 'b') {
145  return '\b';
146  } else if (c == 't') {
147  return '\t';
148  } else if (c == 'n') {
149  return '\n';
150  } else if (c == 'f') {
151  return '\f';
152  } else if (c == 'r') {
153  return '\r';
154  } else if (c == '"') {
155  return '\"';
156  } else if (c == '\'') {
157  return '\'';
158  } else if (c == '\\') {
159  return '\\';
160  } else if (c == 'u') {
161  // \uFFFF -> HEXA
162  if (cps.strArr.length < cps.pos + 4) {
163  // Invalid input
164  throw new JPFInspectorParsingErrorException("Error while parsing character at " + cps.strArr.length
165  + ". Hexa representation of char malformed (\\uFFFF) like character is expected.", new String(cps.strArr), cps.strArr.length);
166  }
167  int hexVal = 0;
168  for (int i = 0; i < 4; i++) {
169  char cx = cps.strArr[cps.pos];
170  cx = Character.toLowerCase(cx);
171 
172  if (cx >= '0' && cx <= '9') {
173  hexVal = (hexVal * 16) + (cx - '0');
174  } else if (cx >= 'a' && cx <= 'f') {
175  hexVal = (hexVal * 16) + (cx - 'a' + 10);
176  } else {
177  // Not e hexa char
178  throw new JPFInspectorParsingErrorException("Error while parsing character at " + (cps.pos)
179  + " - non hexa character found. The \\uFFFF like character is expected.", new String(cps.strArr), cps.pos);
180  }
181  cps.pos++;
182 
183  }
184  return (char) hexVal;
185  } else if (c >= '0' && c <= '7') {
186  // Octal num 0 .. 377 // parsed eagerly
187  int numsLeft = 2; // Maximal number of octal numbers to be processed
188  if (c >= '0' && c <= '3') {
189  numsLeft++;
190  }
191  if (cps.pos + numsLeft > cps.strArr.length) {
192  numsLeft = cps.strArr.length - cps.pos;
193  }
194 
195  assert (numsLeft > 0);
196 
197  int octVal = 0;
198  cps.pos--; // to process actual character in c variable
199  for (int i = 0; i < numsLeft; i++) {
200  char cx = cps.strArr[cps.pos];
201 
202  if (cx >= '0' && cx <= '7') {
203  octVal = (octVal * 8) + (cx - '0');
204  } else {
205  break;
206  }
207 
208  cps.pos++;
209  }
210  return (char) octVal;
211  } else {
212  // invalid escape sequence
213  throw new JPFInspectorParsingErrorException("Error while parsing character at " + (cps.pos - 1) + ". Invalid escape sequence.", new String(cps.strArr),
214  cps.pos - 1);
215  }
216  }

228  public static String unparseChar (char c) {
229  // 1 step - escaping characters
230  // Escaped chars \b \t \n \f \r \" \' \\
231  if (c == '\b') {
232  return "\\b";
233  }
234  if (c == '\t') {
235  return "\\t";
236  }
237  if (c == '\n') {
238  return "\\n";
239  }
240  if (c == '\f') {
241  return "\\f";
242  }
243  if (c == '\"') {
244  return "\\\"";
245  }
246  if (c == '\'') {
247  return "\\\'";
248  }
249  if (c == '\\') {
250  return "\\\\";
251  }
252  // 2 step - convert to unicode represenation
253 
254  if (c < 16) {
255  return "\\u000" + Integer.toHexString(c);
256  }
257  if (c < 32) {
258  return "\\u00" + Integer.toHexString(c);
259  }
260 
261  if (c > 0x0FFF) {
262  return "\\u" + Integer.toHexString(c);
263  }
264  if (c >= 1024) {
265  return "\\u0" + Integer.toHexString(c);
266  }
267 
268  return String.valueOf(c);
269  }
