140  static public Instruction getInstructionForThread (Transition tr, Path path, int threadNum, int instCount) {
141  assert path != null;
142  assert instCount >= 0;
143 
144  int instInThread = 0; // Instructions executed by given thread counter
145  if (tr.getThreadIndex() == threadNum) {
146  instInThread += tr.getStepCount();
147 
148  if (instInThread > instCount) {
149  return tr.getStep(instInThread - instCount - 1).getInstruction();
150  }
151  }
152 
153  for (int i = path.size() - 1; i >= 0; i--) {
154  tr = path.get(i);
155  if (tr.getThreadIndex() != threadNum) {
156  continue; // Different thread was planned
157  }
158 
159  instInThread += tr.getStepCount();
160 
161  if (instInThread > instCount) {
162  return tr.getStep(instInThread - instCount - 1).getInstruction();
163  }
164  }
165 
166  return null;
167  }
