58  static public String breakpointToString (BreakPointStatus bp) {
59  StringBuilder bpText = new StringBuilder(80);
60  bpText.append(bp.getBPID());
61  bpText.append(" : ");
62  if (bp.getName() != null && !bp.getName().equals("")) {
63  bpText.append(" name=");
64  bpText.append(bp.getName());
65  }
66  bpText.append(" state=");
67  bpText.append(bp.getState().toString());
68 
69  bpText.append(" hits=");
70  bpText.append(bp.getHitCounter());
71 
72  bpText.append(" hitsTotal=");
73  bpText.append(bp.getHitCounterTotal());
74 
75  if (bp.bpHitCountLowerBound() != null || bp.bpHitCountUpperBound() != null) {
76  bpText.append(' ');
77  Integer lowerBound = bp.bpHitCountLowerBound();
78  Integer upperBound = bp.bpHitCountUpperBound();
79  if (lowerBound != null && (!lowerBound.equals(BreakPointCreationInformation.DEFAULT_LOWER_BOUND))) {
80  bpText.append(lowerBound);
81  bpText.append("<=");
82  }
83  bpText.append("hit_count");
84  if (upperBound != null && (!upperBound.equals(BreakPointCreationInformation.DEFAULT_UPPER_BOUND))) {
85  bpText.append("<=");
86  bpText.append(upperBound);
87  }
88  }
89 
90  bpText.append(' ');
91  bpText.append(bp.getNormalizedBreakpointExpression());
92 
93  return bpText.toString();
