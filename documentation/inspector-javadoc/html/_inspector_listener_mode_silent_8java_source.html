99  public void stateBacktracked (Search search) {
100  if (DEBUG) {
101  inspector.getDebugPrintStream().println(this.getClass().getSimpleName() + ".stateBacktracked()");
102  }
103  if (state != InternalState.IS_BACKTRACKING && state != InternalState.IS_TRANSITION_END) { // Initial CB is backtracked (advance or processed has been called
104  // before in forward stepping)
105  reportError();
106  return;
107  }
108  inspState.stateChanged(search, ListenerMethod.LM_STATE_BACKTRACKED);
109  backwardStepsCnt--;
110  if (backwardStepsCnt > 0) {
111  dftMgf.extendTrace(search.getTransition());
112 
113  VM vm = search.getVM();
114  // SystemState ss = vm.getSystemState();
115  // ss.setBoring(true); // Force backtracking
116  // Force backtracking
117  search.requestBacktrack();
118 
119  // Disable forward steps - mark all choices from CGs as processed
120  SystemState ss = vm.getSystemState();
121  ChoiceGenerator<?> cg = ss.getChoiceGenerator();
122  while (cg != null) {
123  int choices = cg.getTotalNumberOfChoices();
124  int processedChoices = cg.getProcessedNumberOfChoices();
125  int advanceChoice = choices - processedChoices;
126  cg.advance(advanceChoice);
127  assert (cg.hasMoreChoices() == false);
128  // Go to parent
129  cg = cg.getCascadedParent();
130  }
131  state = InternalState.IS_STATE_PROCESSED;
132  return;
133  }
134  if (backwardStepsCnt == 0) {
135  // Backtracking terminated, use the same choices before
136  VM vm = search.getVM();
137  SystemState ss = vm.getSystemState();
138  ChoiceGenerator<?> cg = ss.getChoiceGenerator();
139 
140  // we must skip choices that have been processed before this call
141  // special case: there might be a cascade with multiple CGs at one instruction
142  // it is not possible to move back a CG in the cascade if there is one processed choice (the first one) -> we must manually "set all choices as being explored", so that this CG is explored again from the start (for the parent choice) on the way forward
143  // this does not apply to the top CG in the cascade (which does not have a cascaded parent)
144  while ((cg.getCascadedParent() != null) && (cg.getProcessedNumberOfChoices() == 1)) {
145  // Set this to last value && end delegate to predecessor
146  cg.advance(cg.getTotalNumberOfChoices() - cg.getProcessedNumberOfChoices());
147  cg = cg.getCascadedParent();
148  }
149  // CG undo last choice
150  int processedChoices = cg.getProcessedNumberOfChoices();
151  assert (processedChoices > 0); // Which choice which used for path we are backtracking??
152  cg.reset();
153  cg.advance(processedChoices - 1);
154  // "Force" forward step
155  ss.setForced(true);
156  state = InternalState.IS_FORWARD_STEPS;
157  return;
158  }
159 
160  // Should not be reached
161  reportError();
162  return;
