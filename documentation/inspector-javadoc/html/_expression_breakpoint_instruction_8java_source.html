83  static ExpressionBreakpointInstruction getBPonPreviousInstruction (Transition tr) {
84  assert tr != null;
85  int instrIndex = tr.getStepCount() - 1;
86  if (instrIndex < 1) {
87  return null;
88  }
89  Step penultimateStep = tr.getStep(instrIndex);
90  Instruction penultimateInstr = penultimateStep.getInstruction();
91 
92  int sameInstrCount = 1; // One for the final instruction where we hit the BP
93  // Count how many time the instruction takes place in the transition (due to cycle repetitions etc.)
94  for (Step currentStep = tr.getStep(0); !currentStep.equals(penultimateStep); currentStep = currentStep.getNext()) {
95  if (penultimateInstr.equals(currentStep.getInstruction())) {
96  sameInstrCount++;
97  }
98  }
99 
100  return new ExpressionBreakpointInstruction(tr.getThreadIndex(), penultimateInstr, sameInstrCount);
101  }

137  public static Instruction findInstruction (JPFInspector inspector, MethodName methodSpec, int instIdx) throws JPFInspectorGenericErrorException {
138  assert inspector != null;
139  StopHolder sh = inspector.getStopHolder();
140  assert sh != null;
141  InspectorState inspState = sh.getInspectorState();
142  if (inspState == null) {
143  throw new JPFInspectorGenericErrorException("Cannot find specified instruction - JPF is not connected to Inspector or execution is not stopped");
144  }
145 
146  VM vm = inspState.getJVM();
147  assert (vm != null);
148 
149  assert (methodSpec != null);
150  ClassName classSpec = methodSpec.getClassNameClass();
151 
152  ClassInfo[] cis = MigrationUtilities.getLoadedClasses();
153  for (ClassInfo ci : cis) {
154  assert (ci != null);
155  if (classSpec.isSameClass(ci)) {
156  MethodInfo[] mis = ci.getDeclaredMethodInfos();
157  for (MethodInfo mi : mis) {
158  if (methodSpec.isSameMethod(mi)) {
159  return mi.getInstruction(instIdx);
160  }
161  }
162  }
163  }
164  // No such method exists
165  return null;
166  }
