52  public static int convertToInteger (String value) throws JPFInspectorParsingErrorException {
53  assert value != null;
54  assert value.length() > 0;
55 
56  String val = value;
57  char c0 = val.charAt(0);
58  if (c0 == '+') {
59  val = val.substring(1); // Ignore initial '+' character
60  }
61  boolean startWithMinus = c0 == '-';
62 
63  // Decide type of the input HEX or INT
64  if (isHEX(val)) {
65  if (startWithMinus) {
66  val = "-" + val.substring(3);
67  } else {
68  val = val.substring(2);
69  }
70  try {
71  return Integer.valueOf(val, 16);
72  } catch (NumberFormatException nfe) {
73  throw new JPFInspectorParsingErrorException("Invalid hex integer value. Value is probably out of the integer range.", value, 0);
