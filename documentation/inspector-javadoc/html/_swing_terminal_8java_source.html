391  @Override
392  public void keyPressed (KeyEvent e) {
393  if (DEBUG) {
394  System.out.println("KeyPressed: " + e + "\n\t e.getKeyCode() ->" + e.getKeyCode() + "\n\te.getModifiers() ->" + e.getModifiers());
395  }
396  int keycode = e.getKeyCode();
397  if (keycode == KeyEvent.VK_LEFT) {
398  if ((e.getModifiers() | KeyEvent.CTRL_MASK) == e.getModifiers()) {
399  // CTRL pressed
400  write(JL_PREV_WORD);
401  e.consume();
402  } else {
403  write(JL_PREV_CHAR);
404  e.consume();
405  }
406  } else if (keycode == KeyEvent.VK_RIGHT) {
407  if ((e.getModifiers() | KeyEvent.CTRL_MASK) == e.getModifiers()) {
408  // CTRL pressed
409  write(JL_NEXT_WORD);
410  e.consume();
411  } else {
412  write(JL_NEXT_CHAR);
413  e.consume();
414  }
415  } else if (keycode == KeyEvent.VK_BEGIN) {
416  write(JL_MOVE_TO_BEG);
417  e.consume();
418  } else if (keycode == KeyEvent.VK_END) {
419  write(JL_MOVE_TO_END);
420  e.consume();
421  } else if (keycode == KeyEvent.VK_DOWN) {
422  write(JL_NEXT_HISTORY);
423  e.consume();
424  } else if (keycode == KeyEvent.VK_UP) {
425  write(JL_PREV_HISTORY);
426  e.consume();
427  } else if (keycode == KeyEvent.VK_INSERT && ((e.getModifiers() | KeyEvent.SHIFT_MASK) == e.getModifiers())) {
428  write(JL_PASTE);
429  e.consume();
430  }
431 
432  try {
433  output.flush();
434  } catch (IOException e1) {
435  }
436  }

442  @Override
443  public void keyTyped (KeyEvent e) {
444  if (DEBUG) {
445  System.out.println("KeyTyped: " + e);
446  }
447  e.consume();
448  if (e.getKeyChar() == ' ' && ((e.getModifiers() | KeyEvent.CTRL_MASK) == e.getModifiers())) {
449  // CTRL+Space -> mapped to TAB to invoke completion
450  write(JL_COMPLETE);
451  } else if (e.getKeyCode() == KeyEvent.VK_BACK_SPACE) {
452  write(JL_DELETE_PREV_CHAR);
453  } else if (e.getKeyCode() == KeyEvent.VK_DELETE) {
454  write(JL_DELETE_NEXT_CHAR);
455  } else {
456  write(e.getKeyChar());
457  }
458 
459  try {
460  output.flush();
461  } catch (IOException e1) {
462  }
463  }
