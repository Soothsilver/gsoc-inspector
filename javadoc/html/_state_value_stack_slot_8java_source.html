97  private static StateValueStackSlot createSVSSInstance(StateStackFrame ssf, int referenceDepth,
98  String stateExpression, int slotIndex)
99  throws JPFInspectorInvalidSlotIndexException {
100 
101  assert (ssf != null);
102 
103  StackFrame sf = ssf.getStackFrame();
104 
105  // Check if index is in range
106  final int maxSlotIndex = sf.getTopPos();
107 
108  if (slotIndex < 0 || slotIndex > maxSlotIndex) {
109  throw new JPFInspectorInvalidSlotIndexException(slotIndex, sf.getMethodInfo(), maxSlotIndex);
110  }
111 
112  LocalVarInfo lvi = sf.getLocalVarInfo(slotIndex);
113  if (lvi == null) {
114  // Created faked (raw) LocalVarInfo - represents int
115  lvi = new LocalVarInfo("???-RawView-no Name or Type provided", "I", "I", 0, sf.getMethodInfo().getLastInsn().getPosition(), slotIndex);
116 
117  }
118  assert lvi != null;
119 
120  // Obtain real type of the field
121  String className = Types.getTypeName(lvi.getSignature());
122  if (DEBUG) {
123  ssf.getInspector().getDebugPrintStream().println(StateValueStackSlot.class.getSimpleName() + ".createSVSSInstance - className=" + className);
124  }
125  // Can throw NoClassInfoException -> in such a case it is internal error
126  // An alternative here is to use getSystemResolvedClassInfo which uses the system class loader rather than the
127  // thread's current class loader. I don't know the difference between the two so we'll use the current one for now,
128  // and see what develops.
129  // The original pre-migration source line was:
130  // ClassInfo ciReal = ClassInfo.getResolvedClassInfo(className);
131  ClassInfo ciReal = ClassLoaderInfo.getCurrentResolvedClassInfo(className);
132 
133  assert (ciReal != null);
134  assert (StateValue.isPredecessor(ciReal, ciReal));
135 
136  return new StateValueStackSlot(ssf, referenceDepth, stateExpression, ciReal, slotIndex, lvi);
