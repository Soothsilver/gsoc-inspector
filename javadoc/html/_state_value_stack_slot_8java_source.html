101  private static StateValueStackSlot createSVSSInstance(StateStackFrame ssf,
102  String stateExpression,
103  int slotIndex)
104  throws JPFInspectorInvalidSlotIndexException {
105 
106  assert (ssf != null);
107 
108  StackFrame sf = ssf.getStackFrame();
109 
110  // Check if index is in range
111  final int maxSlotIndex = sf.getTopPos();
112 
113  if (slotIndex < 0 || slotIndex > maxSlotIndex) {
114  throw new JPFInspectorInvalidSlotIndexException(slotIndex, sf.getMethodInfo(), maxSlotIndex);
115  }
116 
117  LocalVarInfo lvi = sf.getLocalVarInfo(slotIndex);
118  if (lvi == null) {
119  // Created faked (raw) LocalVarInfo - represents int
120  lvi = new LocalVarInfo("???-RawView-no Name or Type provided", "I", "I", 0, sf.getMethodInfo().getLastInsn().getPosition(), slotIndex);
121 
122  }
123  assert lvi != null;
124 
125  // Obtain real type of the field
126  String className = Types.getTypeName(lvi.getSignature());
127  if (DEBUG) {
128  ssf.getInspector().getDebugPrintStream().println(StateValueStackSlot.class.getSimpleName() + ".createSVSSInstance - className=" + className);
129  }
130  // Can throw NoClassInfoException -> in such a case it is internal error
131  // An alternative here is to use getSystemResolvedClassInfo which uses the system class loader rather than the
132  // thread's current class loader. I don't know the difference between the two so we'll use the current one for now,
133  // and see what develops.
134  // The original pre-migration source line was:
135  // ClassInfo ciReal = ClassInfo.getResolvedClassInfo(className);
136  ClassInfo ciReal = ClassLoaderInfo.getCurrentResolvedClassInfo(className);
137 
138  assert (ciReal != null);
139  assert (StateWritableValue.isPredecessor(ciReal, ciReal));
140 
141  return new StateValueStackSlot(ssf, stateExpression, ciReal, slotIndex, lvi);
142 
143  }
