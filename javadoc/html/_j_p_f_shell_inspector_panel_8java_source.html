128  @Override
129  public void run () {
130  try {
131  //noinspection InfiniteLoopStatement
132  while (true) {
133  try {
134  String s;
135  if (initialCmdProcessed < INITIAL_COMMANDS.length) {
136  // if (initialCmdProcessed > 1 && INITIAL_COMMANDS[initialCmdProcessed-1].equals("run")) sleep(500);
137  // useful for debugging (if some commands are predefined)
138  sleep(1500);
139  s = INITIAL_COMMANDS[initialCmdProcessed++];
140  } else {
141  s = console.readLine(Constants.PROMPT);
142  }
143  String sTrimmed = s.trim();
144  if (sTrimmed.length() == 0) {
145  consolePrintStream.println("\n" + Constants.PROMPT);
146  }
147  if (s.trim().length() > 0) {
148  consolePrintStream.println();
149  inspector.executeCommand(s, ExecutionContext.FROM_SWING_TERMINAL);
150  }
151  } catch (IOException e) {
152  consolePrintStream.println("ERR: Error while reading a command.");
153  }
154  } // end of while
155  } catch (InterruptedException e) {
156  // Terminate the command loop
157  }
158  }
159  }
