137  public static BackwardBreakpointCreator getBackwardStepInstruction (InspectorState inspectorState) {
138  Path path = updateAndGetPath(inspectorState);
139  Transition currentTransition = path.getLast();
140  if (currentTransition == null) {
141  // Called too early,
142  assert false; // This should not happen.
143  return null;
144  }
145  int currentThread = currentTransition.getThreadIndex();
146 
147  StepThreadBacktracker stb = new StepThreadBacktracker(new TransitionThreadBacktracker(path, currentThread));

154  }
155

175  public static BackwardBreakpointCreator getBackwardStepLine (InspectorState inspectorState) {
176  Path path = updateAndGetPath(inspectorState);
177  MethodInstructionBacktracker methodInstructionBacktracker = new MethodInstructionBacktracker(path);
178 
179  Instruction currentInstruction = inspectorState.getVM().getInstruction();
180  InstructionPosition currentLocation = InstructionPositionImpl.getInstructionPosition(currentInstruction);
181  Debugging.getLogger().info("Determining target instruction: back_step_over first step: " + currentInstruction + " at line " + currentLocation.getLineNumber());
182 
183  // First step back.
184  Step step = methodInstructionBacktracker.backtrackToPreviousStepInMethod();
185 
186  // Undo instructions on the current line
187  while ((step != null) && currentLocation.hitPosition(step.getInstruction())) {
188  step = methodInstructionBacktracker.backtrackToPreviousStepInMethod();
189  }
190 
191  if (step == null) {
192  return null; // We have reached the beginning of the thread or the method.
193  }
194 
195  // We are now either on the previous line or in the caller.
196  if (step.getInstruction().getMethodInfo() != currentInstruction.getMethodInfo()) {
197  // We are in the caller - this happened because we were already at the first line of a method.
198  // However, that does not matter. For consistency, we will backtrack to the beginning of the line anyway.
199  }
200 
201 
202  // We are on the previous line - stop after the first instruction on given line
203  InstructionPosition previousLinePosition = InstructionPositionImpl.getInstructionPosition(step.getInstruction());
204 
205  // Be careful - we need to remember previous step and transition2backreack
206  Step previousStep = null;
207  Transition previousTransition = null;
208  int previousTransitionsToBacktrack = 0;
209 
210  while (step != null && previousLinePosition.hitPosition(step.getInstruction())) {
211  previousStep = step;
212  previousTransition = methodInstructionBacktracker.getCurrentTransition();
213  previousTransitionsToBacktrack = methodInstructionBacktracker.getBacktrackedTransitionCount();
214  step = methodInstructionBacktracker.backtrackToPreviousStepInMethod();
215  }
216 
217  // previousStep now contains the first step on the previous code line
218  // previousTransition and previousTransitionsToBacktrack hold information relating to previousStep
219  // Yes, we could have named it better, as this confuses the meaning of "previous' as in "previous code line"
220  // and "previous" as in "the step we considered before but is actually after"
221  return new BackwardBreakpointCreator(previousTransition, previousStep, previousTransitionsToBacktrack);
222  }

307  public static BackwardBreakpointCreator getBackBreakpointHit(BreakpointHitLocation lastBreakpointHitLocation,
308  InspectorState inspState)
309  throws JPFInspectorGenericErrorException {
310  if (lastBreakpointHitLocation == null) {
311  throw new JPFInspectorGenericErrorException("No breakpoint was yet hit.");
312  }
313  inspState.getVM().updatePath();
314  Path path = inspState.getVM().getPath();
315  int targetTransitionIndex = lastBreakpointHitLocation.getIndexOfTransitionInPath();
316  if (path.size() <= targetTransitionIndex) {
317  throw new JPFInspectorGenericErrorException("You have already backtracked through the transition where the last breakpoint was hit (not enough transitions).");
318  }
319  Transition targetTransition = path.get(targetTransitionIndex);
320 
321  int instructionsToSkip = lastBreakpointHitLocation.getNumberOfSkippedInstructions();
322  Step targetStep = null;
323  for (Step step : targetTransition) {
324  if (step.getInstruction().equals(lastBreakpointHitLocation.getInstruction())) {
325  if (instructionsToSkip == 0) {
326  targetStep = step;
327  break;
328  } else {
329  instructionsToSkip--;
330  }
331  }
332  }
333  if (targetStep == null) {
334  throw new JPFInspectorGenericErrorException("You have already backtracked through the instruction where the last breakpoint was hit (target instruction not found).");
335  }
336 
337  int descendHowManyTransitions = path.size() - targetTransitionIndex;
338  assert descendHowManyTransitions > 0;
339  BackwardBreakpointCreator backwardBreakpointCreator =
340  new BackwardBreakpointCreator(targetTransition, lastBreakpointHitLocation.getInstruction(),
341  lastBreakpointHitLocation.getNumberOfSkippedInstructions(),
342  descendHowManyTransitions);
343  return backwardBreakpointCreator;
344  }

346  public static BackwardBreakpointCreator getBackwardStepTransition(InspectorState inspState,
347  CommandsInterface.StepType type,
348  int descendHowManyTransitions)
349  throws JPFInspectorGenericErrorException {
350  assert descendHowManyTransitions >= 1;
351 
352  Path path = updateAndGetPath(inspState);
353  Transition targetTransition = null;
354  int remainingTransitionsToSkip = descendHowManyTransitions - 1;
355  for (int i = path.size() - 1; i >= 0; i--) {
356  Transition possibleTargetTransition = path.get(i);
357  boolean correctType;
358  switch (type) {
359  case ST_TRANSITION_ALL: correctType = true; break;
360  case ST_TRANSITION_SCHED:
361  correctType = possibleTargetTransition.getChoiceGenerator() instanceof ThreadChoiceGenerator; break;
362  case ST_TRANSITION_DATA:
363  correctType = !(possibleTargetTransition.getChoiceGenerator() instanceof ThreadChoiceGenerator); break;
364  default:
365  throw new JPFInspectorGenericErrorException("This step type is not legal at this point.");
366  }
367  if (correctType) {
368  if (remainingTransitionsToSkip == 0) {
369  targetTransition = possibleTargetTransition;
370  break;
371  } else {
372  remainingTransitionsToSkip--;
373  }
374  }
375  }
376  if (targetTransition == null) {
377  throw new JPFInspectorGenericErrorException(
378  "There aren't that many transitions beginning with the specified choice generator type in the transition path.");
379  }
380 
381  if (targetTransition.getStepCount() <= 0) {
382  if (descendHowManyTransitions == 1) {
383  throw new JPFInspectorGenericErrorException("You are already at the beginning of this transaction. Use 'back_step_transition 2' to backtrack further.");
384  }
385  throw new JPFInspectorGenericErrorException("The target transition has no steps and cannot be backtracked to.");
386  }
387  BackwardBreakpointCreator bbc = new BackwardBreakpointCreator(targetTransition,
388  targetTransition.getStep(0),
389  descendHowManyTransitions);
390  return bbc;
391  }

401  public static BackwardBreakpointCreator getBackwardFieldAccess(FieldName fieldName,
402  InspectorState inspectorState)
403  throws JPFInspectorGenericErrorException {
404  Path path = updateAndGetPath(inspectorState);
405  Transition targetTransition = null;
406  int descendHowManyTransitions = 0;
407  for (int i = path.size() - 1; i >= 0; i--) {
408  Transition possibleTargetTransition = path.get(i);
409 
410  if (possibleTargetTransition.getStepCount() <= 0) {
411  // This usually means that it's the CURRENT transition, i.e. we are just before the bottom half of the instruction
412  // that caused this transition. In that case, we backtrack.
413  continue;
414  }
415  Instruction firstInstruction = possibleTargetTransition.getStep(0).getInstruction();
416  if (firstInstruction instanceof FieldInstruction) {
417  FieldInfo fieldInfo = ((FieldInstruction)firstInstruction).getFieldInfo();
418  if (fieldName.isSameField(fieldInfo)) {
419  // Bingo!
420  targetTransition = possibleTargetTransition;
421  descendHowManyTransitions = path.size() - i;
422  break;
423  } else {
424  // It's a different field.
425  }
426  } else {
427  // We are only interested in field instructions.
428  }
429  }
430  if (targetTransition == null) {
431  throw new JPFInspectorGenericErrorException(
432  "No field matching the argument is accessed in the transition path.");
433  }
434  assert targetTransition.getStepCount() > 0;
435  BackwardBreakpointCreator bbc = new BackwardBreakpointCreator(targetTransition,
436  targetTransition.getStep(0),
437  descendHowManyTransitions);
438  return bbc;
439  }
