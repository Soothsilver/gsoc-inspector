81  private void changeBreakpointStatus(BreakpointState newState) {
82  java.util.List<BreakpointStatus> statuses = breakpointStatusJList.getSelectedValuesList();
83  for (BreakpointStatus bpStatus : statuses) {
84  inspectorClient.getServer().changeBreakpointState(bpStatus.getBPID(), newState);

90  @SuppressWarnings({"UnusedParameters", "SameParameterValue"})
91  private void listSelectionChanged(ListSelectionEvent listSelectionEvent) {
92  java.util.List<BreakpointStatus> statuses = breakpointStatusJList.getSelectedValuesList();
93  boolean canEnable = false;
94  boolean canLog = false;
95  boolean canDisable = false;
96  boolean canDelete = false;
97  for (BreakpointStatus status : statuses) {
98  canDelete = true;
99  switch (status.getState()) {
100  case DISABLED: canEnable = true; canLog = true; break;
101  case ENABLED: canDisable = true; canLog = true; break;
102  case LOGGING: canEnable = true; canDisable = true; break;
103  }
104  }
105  buttonEnable.setEnabled(canEnable);
106  buttonDelete.setEnabled(canDelete);
107  buttonDisable.setEnabled(canDisable);
108  buttonLogging.setEnabled(canLog);
109  }
