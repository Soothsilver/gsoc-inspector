49  public static String elementInfo2String(ElementInfo ei) {
50  if (ei == null) {
51  return "null";
52  }
53 
54  if (ei.isArray()) {
55  // Note: Bad code practice - alternative way,
56  // Create Decorating factory, decorators with used defined toSting() methods
57  // Update configuration- "vm.fields_factory.class
58 
59  ClassInfo elemClassInfo = ei.getClassInfo().getComponentClassInfo();
60  assert elemClassInfo != null;
61  String elemSignature = elemClassInfo.getSignature();
62 
63  if ("Z".equals(elemSignature)) {
64  return Arrays.toString(ei.asBooleanArray());
65  } else if ("B".equals(elemSignature)) {
66  return Arrays.toString(ei.asByteArray());
67  } else if ("C".equals(elemSignature)) {
68  return Arrays.toString(ei.asCharArray());
69  } else if ("S".equals(elemSignature)) {
70  return Arrays.toString(ei.asShortArray());
71  } else if ("I".equals(elemSignature)) {
72  return Arrays.toString(ei.asIntArray());
73  } else if ("J".equals(elemSignature)) {
74  return Arrays.toString(ei.asLongArray());
75  } else if ("F".equals(elemSignature)) {
76  return Arrays.toString(ei.asFloatArray());
77  } else if ("D".equals(elemSignature)) {
78  return Arrays.toString(ei.asDoubleArray());
79  }
80  if (elemSignature != null && elemSignature.startsWith("L") || elemClassInfo.isArray()) {
81  int[] refArr = ei.asReferenceArray();
82  StringBuilder sb = new StringBuilder();
83  sb.append('[');
84  for (int i = 0; i < refArr.length; i++) {
85  if (i > 0) {
86  sb.append(", ");
87  }
88  if (refArr[i] == MJIEnv.NULL) {
89  sb.append("null");
90  } else {
91  // value stored in array at index i
92  ElementInfo eiArr = VM.getVM().getHeap().get(refArr[i]);
93  assert (eiArr != null);
94 
95  sb.append(elementInfo2String(eiArr));
96  }
97  }
98  sb.append(']');
99  return sb.toString();
100  } else {
101  throw new RuntimeException("Invalid signature " + elemSignature);
102  }
103  }
104  if (ClassInfo.isStringClassInfo(ei.getClassInfo()) && (ei instanceof DynamicElementInfo)) {
105  DynamicElementInfo dei = (DynamicElementInfo) ei;
106  return '\"' + dei.asString() + '\"';
107  }
108 
109  // Normally, we would have ElementInfo.toString here. However, we want to display the heap index
110  // as a decimal value, not hexadecimal.
111  // return ((ci != null ? ci.getName() : "ElementInfo") + '@' + Integer.toHexString(objRef));
112  String className = (ei.getClassInfo()!= null ? ei.getClassInfo().getName() : "ElementInfo");
113  String heapIndex = Integer.toString(ei.getObjectRef(), 10);
114  return className + "@" + heapIndex;
115  }
