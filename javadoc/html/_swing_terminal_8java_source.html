395  @Override
396  public void keyPressed (KeyEvent e) {
397  if (DEBUG) {
398  System.out.println("KeyPressed: " + e + "\n\t e.getKeyCode() ->" + e.getKeyCode() + "\n\te.getModifiers() ->" + e.getModifiers());
399  }
400  int keycode = e.getKeyCode();
401  if (keycode == KeyEvent.VK_LEFT) {
402  if ((e.getModifiers() | KeyEvent.CTRL_MASK) == e.getModifiers()) {
403  // CTRL pressed
404  write(JL_PREV_WORD);
405  e.consume();
406  } else {
407  write(JL_PREV_CHAR);
408  e.consume();
409  }
410  } else if (keycode == KeyEvent.VK_RIGHT) {
411  if ((e.getModifiers() | KeyEvent.CTRL_MASK) == e.getModifiers()) {
412  // CTRL pressed
413  write(JL_NEXT_WORD);
414  e.consume();
415  } else {
416  write(JL_NEXT_CHAR);
417  e.consume();
418  }
419  } else if (keycode == KeyEvent.VK_BEGIN) {
420  write(JL_MOVE_TO_BEG);
421  e.consume();
422  } else if (keycode == KeyEvent.VK_END) {
423  write(JL_MOVE_TO_END);
424  e.consume();
425  } else if (keycode == KeyEvent.VK_DOWN) {
426  write(JL_NEXT_HISTORY);
427  e.consume();
428  } else if (keycode == KeyEvent.VK_UP) {
429  write(JL_PREV_HISTORY);
430  e.consume();
431  } else if (keycode == KeyEvent.VK_INSERT && ((e.getModifiers() | KeyEvent.SHIFT_MASK) == e.getModifiers())) {
432  write(JL_PASTE);
433  e.consume();
434  }
435 
436  try {
437  output.flush();
438  } catch (IOException ignored) {
439  }
440  }

446  @Override
447  public void keyTyped (KeyEvent e) {
448  if (DEBUG) {
449  System.out.println("KeyTyped: " + e);
450  }
451  e.consume();
452  if (e.getKeyChar() == ' ' && ((e.getModifiers() | KeyEvent.CTRL_MASK) == e.getModifiers())) {
453  // CTRL+Space -> mapped to TAB to invoke completion
454  write(JL_COMPLETE);
455  } else if (e.getKeyCode() == KeyEvent.VK_BACK_SPACE) {
456  write(JL_DELETE_PREV_CHAR);
457  } else if (e.getKeyCode() == KeyEvent.VK_DELETE) {
458  write(JL_DELETE_NEXT_CHAR);
459  } else {
460  write(e.getKeyChar());
461  }
462 
463  try {
464  output.flush();
465  } catch (IOException ignored) {
466  }
467  }
