401  @Override
402  public void keyPressed (KeyEvent e) {
403  if (DEBUG) {
404  System.out.println("KeyPressed: " + e + "\n\t e.getKeyCode() ->" + e.getKeyCode() + "\n\te.getModifiers() ->" + e.getModifiers());
405  }
406  int keycode = e.getKeyCode();
407  if (keycode == KeyEvent.VK_LEFT) {
408  if ((e.getModifiers() | KeyEvent.CTRL_MASK) == e.getModifiers()) {
409  // CTRL pressed
410  write(JL_PREV_WORD);
411  e.consume();
412  } else {
413  write(JL_PREV_CHAR);
414  e.consume();
415  }
416  } else if (keycode == KeyEvent.VK_RIGHT) {
417  if ((e.getModifiers() | KeyEvent.CTRL_MASK) == e.getModifiers()) {
418  // CTRL pressed
419  write(JL_NEXT_WORD);
420  e.consume();
421  } else {
422  write(JL_NEXT_CHAR);
423  e.consume();
424  }
425  } else if (keycode == KeyEvent.VK_BEGIN) {
426  write(JL_MOVE_TO_BEG);
427  e.consume();
428  } else if (keycode == KeyEvent.VK_END) {
429  write(JL_MOVE_TO_END);
430  e.consume();
431  } else if (keycode == KeyEvent.VK_DOWN) {
432  write(JL_NEXT_HISTORY);
433  e.consume();
434  } else if (keycode == KeyEvent.VK_UP) {
435  write(JL_PREV_HISTORY);
436  e.consume();
437  } else if (keycode == KeyEvent.VK_INSERT && ((e.getModifiers() | KeyEvent.SHIFT_MASK) == e.getModifiers())) {
438  write(JL_PASTE);
439  e.consume();
440  }
441 
442  try {
443  output.flush();
444  } catch (IOException ignored) {
445  }
446  }

452  @Override
453  public void keyTyped (KeyEvent e) {
454  if (DEBUG) {
455  System.out.println("KeyTyped: " + e);
456  }
457  e.consume();
458  if (e.getKeyChar() == ' ' && ((e.getModifiers() | KeyEvent.CTRL_MASK) == e.getModifiers())) {
459  // CTRL+Space -> mapped to TAB to invoke completion
460  write(JL_COMPLETE);
461  } else if (e.getKeyCode() == KeyEvent.VK_BACK_SPACE) {
462  write(JL_DELETE_PREV_CHAR);
463  } else if (e.getKeyCode() == KeyEvent.VK_DELETE) {
464  write(JL_DELETE_NEXT_CHAR);
465  } else {
466  write(e.getKeyChar());
467  }
468 
469  try {
470  output.flush();
471  } catch (IOException ignored) {
472  }
473  }
