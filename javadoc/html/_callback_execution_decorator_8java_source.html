117  public WORKING_MODE setNewMode (WORKING_MODE newMode) {
118  synchronized (syncObj) {
119  WORKING_MODE oldMode = mode;
120  if (newMode != null) {
121  mode = newMode;
122 
123  // Unblock blocked callbacks (and/or Command thread)
124  if (newMode.equals(WORKING_MODE.WM_USER_COMMANDS)) {
125  syncObj.notifyAll();
126  }
127 
128  if (oldMode == WORKING_MODE.WM_USER_COMMANDS && newMode == WORKING_MODE.WM_EXECUTION_RECORD) {

130  }
131  }
132  return oldMode;

136  // Block CB thread until Command thread specifies which CB is expected
137  // Null means pass any possible callback
138  private void waitUntilCBIsSpecified(CallbackKind calledCallback) {
139  synchronized (syncObj) {
140  if (mode == WORKING_MODE.WM_EXECUTION_RECORD) {
141 
142  while (method == null && mode == WORKING_MODE.WM_EXECUTION_RECORD) {
143  isCBwaiting = true;
144  try {
145  syncObj.wait();
146  } catch (InterruptedException e) {
147  // Ignore
148  }
149  isCBwaiting = false;
150  }
151 
152  if (method != null && mode == WORKING_MODE.WM_EXECUTION_RECORD) {
153  // Null means any callback is possible
154  if (!(CallbackKind.CB_ANY.equals(calledCallback) || method.equals(calledCallback))) {
155  // Set error state
156  }
157  }
158  }
159  }
160  }

170  private void waitForCB(CallbackKind wait4method) {
171  synchronized (syncObj) {
172  if (mode == WORKING_MODE.WM_EXECUTION_RECORD) {
173 
174  // Wait until we either leave record execution mode or until all wait-for-callback commands are processed.
175  // In practice, I think this loop shouldn't ever be entered, but what do I know.
176  while (method != null && mode == WORKING_MODE.WM_EXECUTION_RECORD) {
177  try {
178  syncObj.wait();
179  } catch (InterruptedException e) {
180  Thread.currentThread().interrupt();
181  }
182  }
183 
184  // If, in the meantime, we left record execution mode, then this command ceases to have any effect.
185  if (mode == WORKING_MODE.WM_EXECUTION_RECORD) {
186  method = wait4method;
187 
188  if (isCBwaiting) {
189  // Wake up waiting CB thread
190  syncObj.notifyAll();
191  }
192 
193  // Wait until CB processing is terminated
194  try {
195  syncObj.wait();
196  } catch (InterruptedException e) {
197  Thread.currentThread().interrupt();
198  }
199  }
200  }
201  }
202  }

204  // Called by CB thread to wake up blocked Command thread
205  private void unblockCmdThread() {
206  synchronized (syncObj) {
207  assert (isCBwaiting == false);
208  method = null;
209  syncObj.notifyAll();
210  }
