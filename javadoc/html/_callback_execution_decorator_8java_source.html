127  public WORKING_MODE setNewMode (WORKING_MODE newMode) {
128  synchronized (syncObj) {
129  WORKING_MODE oldMode = mode;
130  if (newMode != null) {
131  mode = newMode;
132 
133  // Unblock blocked callbacks (and/or Command thread)
134  if (newMode.equals(WORKING_MODE.WM_USER_COMMANDS)) {
135  syncObj.notifyAll();
136  }
137 
138  if (oldMode == WORKING_MODE.WM_USER_COMMANDS && newMode == WORKING_MODE.WM_EXECUTION_RECORD) {
139  // Start new replay of execution record -> clear errors
140  error = false;
141  }
142  }
143  return oldMode;
144  }
145  }

147  // Block CB thread until Command thread specifies which CB is expected
148  // Null means pass any possible callback
149  private void waitUntilCBIsSpecified(CallbackKind calledCallback) {
150  synchronized (syncObj) {
151  if (mode == WORKING_MODE.WM_EXECUTION_RECORD) {
152 
153  while (method == null && mode == WORKING_MODE.WM_EXECUTION_RECORD) {
154  isCBwaiting = true;
155  try {
156  syncObj.wait();
157  } catch (InterruptedException e) {
158  // Ignore
159  }
160  isCBwaiting = false;
161  }
162 
163  if (method != null && mode == WORKING_MODE.WM_EXECUTION_RECORD) {
164  // Null means any callback is possible
165  if (!(CallbackKind.CB_ANY.equals(calledCallback) || method.equals(calledCallback))) {
166  // Set error state
167  error = true;
168  }
169  }
170  }
171  }
172  }

182  private void waitForCB(CallbackKind wait4method) {
183  synchronized (syncObj) {
184  if (mode == WORKING_MODE.WM_EXECUTION_RECORD) {
185 
186  // Wait until we either leave record execution mode or until all wait-for-callback commands are processed.
187  // In practice, I think this loop shouldn't ever be entered, but what do I know.
188  while (method != null && mode == WORKING_MODE.WM_EXECUTION_RECORD) {
189  try {
190  syncObj.wait();
191  } catch (InterruptedException e) {
192  Thread.currentThread().interrupt();
193  }
194  }
195 
196  // If, in the meantime, we left record execution mode, then this command ceases to have any effect.
197  if (mode == WORKING_MODE.WM_EXECUTION_RECORD) {
198  method = wait4method;
199 
200  if (isCBwaiting) {
201  // Wake up waiting CB thread
202  syncObj.notifyAll();
203  }
204 
205  // Wait until CB processing is terminated
206  try {
207  syncObj.wait();
208  } catch (InterruptedException e) {
209  Thread.currentThread().interrupt();
210  }
211  }
212  }
213  }
214  }

216  // Called by CB thread to wake up blocked Command thread
217  private void unblockCmdThread() {
218  synchronized (syncObj) {
219  assert (isCBwaiting == false);
220  method = null;
221  syncObj.notifyAll();
222  }
