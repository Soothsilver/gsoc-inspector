56  public static StateValueArrayElement createArrayElement (StateReadableValueInterface srvi, int elementIndex, int referenceDepth) throws JPFInspectorException {
57  assert (srvi != null);
58 
59  ClassInfo arrayCi = srvi.getClassInfo();
60  if (!arrayCi.isArray()) {
61  throw new JPFInspectorNotArrayException(arrayCi);
62  }
63 
64  ElementInfo ei = srvi.getReferenceValue();
65  if (ei == null) {
66  throw new JPFInspectorNullValueException(elementIndex, arrayCi);
67  }
68 
69  // Check array length
70  final Fields fields = ei.getFields();
71  assert (fields instanceof ArrayFields);
72 
73  final ArrayFields afields = (ArrayFields) fields;
74 
75  final int arrayLen = afields.arrayLength();
76 
77  if (elementIndex < 0 || elementIndex >= arrayLen) {
78  throw new JPFInspectorArrayIndexOutOutRangeException(elementIndex, arrayLen, arrayCi);
79  }
80 
81  return new StateValueArrayElement(srvi, referenceDepth, arrayCi.getComponentClassInfo(), srvi.getStateExpr() + '[' + elementIndex + ']', elementIndex, ei,
82  arrayCi);
83  }
