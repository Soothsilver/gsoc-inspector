75  private void loadAttributeAdaptors(Config config) {
76  Class<?>[] adaptors = config.getClasses("jpf-inspector.attribute_adaptors");
77  for (Class<?> adaptorClass : adaptors) {
78  Object adaptorInstance = instantiateClass(adaptorClass);
79  if (adaptorInstance == null) {
80  continue;
81  }
82  boolean isAtLeastOne = false;
83  if (adaptorInstance instanceof AttributeToStringConverter) {
84  isAtLeastOne = true;
85  attributeToStringConverters.add((AttributeToStringConverter)adaptorInstance);
86  }
87  if (adaptorInstance instanceof StringToAttributeConverter) {
88  isAtLeastOne = true;
89  stringToAttributeConverters.add((StringToAttributeConverter)adaptorInstance);
90  }
91  if (adaptorInstance instanceof AttributeAccessDetector) {
92  isAtLeastOne = true;
93  attributeAccessDetectors.add((AttributeAccessDetector)adaptorInstance);
94  }
95  if (!isAtLeastOne) {
96  logger.warning("Class '" + adaptorClass.getName() + "' does not implement any of the attribute adaptor interfaces " +
97  "and thus it will not be loaded.");
98  }
99  }
100  }

113  private void loadCustomHitConditions(Config config) {
114  customHitConditions = new HashMap<>();
115  Class<?>[] customHitConditionClasses = config.getClasses("jpf-inspector.custom_hit_conditions");
116  for(Class<?> customClass : customHitConditionClasses) {
117  Object exampleInstance = instantiateClass(customClass);
118  if (exampleInstance == null) {
119  continue;
120  }
121  if (!(exampleInstance instanceof CustomHitCondition)) {
122  logger.warning("Class '" + customClass.getName() + "' does not implement the CustomHitCondition interface. It won't be usable in the Inspector.");
123  continue;
124  }
125  CustomHitCondition customHitCondition = (CustomHitCondition) exampleInstance;
126  for (String name : customHitCondition.getNames()) {
127  customHitConditions.put(name, customHitCondition.getClass());
128  }
129  }

151  private Object instantiateClass(Class<?> classToInstantiate) {
152  Object newInstance;
153  try {
154  newInstance = classToInstantiate.newInstance();
155  } catch (InstantiationException e) {
156  logger.warning("Class '" + classToInstantiate.getName() + "' cannot be instantiated because of the InstantiationException: " + e.toString() + ". It won't be usable in the Inspector.");
157  return null;
158  } catch (IllegalAccessException e) {
159  logger.warning("Class '" + classToInstantiate.getName() + "' cannot be instantiated because of the IllegalAccessException: " + e.toString()+ ". It won't be usable in the Inspector.");
160  return null;
161  }
162  return newInstance;
163  }

210  public boolean isClassIgnored(String className) {

220  return true;
221  }
222  }
223  return false;
