61  Instruction currentInstruction = inspectorClient.getServer().getCurrentInstruction();
62  if (currentInstruction == null) {
63  activeInstructionChanged("???", null, -1, null);
64  return;
65  }
66  MethodInfo methodInfo = currentInstruction.getMethodInfo();
67  if (methodInfo == null) {
68  activeInstructionChanged(currentInstruction.getMnemonic(), null, currentInstruction.getLineNumber(), null);
69  return;
70  }
71  ClassInfo classInfo = methodInfo.getClassInfo();
72  String sourceFileName = classInfo.getSourceFileName();
73  if (sourceFileName == null) {
74  activeInstructionChanged(currentInstruction.getMnemonic(), null, currentInstruction.getLineNumber(), null);
75  return;
76  }
77  Source source = Source.getSource(sourceFileName);
78  if (source == null) {
79  activeInstructionChanged(currentInstruction.getMnemonic(), sourceFileName, currentInstruction.getLineNumber(), null);
80  return;
81  }
82  StringBuilder sourceBuilder = new StringBuilder();
83  int lineCount = source.getLineCount();
84  for (int i = 1; i <= lineCount; i++) {
85  String thisLine= source.getLine(i);
86  sourceBuilder.append(thisLine).append("\n");
87  }
88  activeInstructionChanged(currentInstruction.getMnemonic(), sourceFileName, currentInstruction.getLineNumber(), sourceBuilder.toString());
89  }
