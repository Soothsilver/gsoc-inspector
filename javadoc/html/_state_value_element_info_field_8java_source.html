159  public static StateReadableValue createOuterClass (StateReadableValue srvi) throws JPFInspectorException {
160  assert (srvi != null);
161 
162  ClassInfo ci = srvi.getClassInfo();
163  ClassInfo ciOuter = ci.getEnclosingClassInfo();
164  if (ciOuter == null) {
165  throw new JPFInspectorNotInnerClassException(ci);
166  }
167 
168  String newStateExpr = srvi.getStateExpr() + "." + PSEVariable.EXPRESSION_OUTER_CLASS;
169  ElementInfo ei = srvi.getReferenceValue();
170  if (StateWritableValue.isStaticElementInfo(ei)) {
171  return StateElementInfo.createStaticOuterClass(srvi);
172  }
173  // Outer class can exists
174  if (ei == null) {
175  throw new JPFInspectorNotInnerClassException(ci);
176  }
177  // Implementation based on {@link DynamicElementInfo#getEnclosingElementInfo()}
178  for (FieldInfo fi : ci.getDeclaredInstanceFields()) {
179  // Note name can be compiler dependent
180  // after the '$' is number representing depth of nesting
181  if (fi.getName().startsWith("this$")) {
182  return new StateValueElementInfoField(srvi, srvi.shouldExpandMembers(), newStateExpr, ei, fi);
183  }
184  }
185 
186  assert false; // Should be have outer class but not Field "this$"num not exists
187  throw new JPFInspectorNotInnerClassException(ci);
188  }
