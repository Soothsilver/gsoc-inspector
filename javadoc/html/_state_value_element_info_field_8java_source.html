157  public static StateReadableValue createOuterClass (StateReadableValue srvi) throws JPFInspectorException {
158  assert (srvi != null);
159 
160  ClassInfo ci = srvi.getClassInfo();
161  ClassInfo ciOuter = ci.getEnclosingClassInfo();
162  if (ciOuter == null) {
163  throw new JPFInspectorNotInnerClassException(ci);
164  }
165 
166  String newStateExpr = srvi.getStateExpr() + "." + PSEVariable.EXPRESSION_OUTER_CLASS;
167  ElementInfo ei = srvi.getReferenceValue();
168  if (StateWritableValue.isStaticElementInfo(ei)) {
169  return StateElementInfo.createStaticOuterClass(srvi);
170  }
171  // Outer class can exists
172  if (ei == null) {
173  throw new JPFInspectorNotInnerClassException(ci);
174  }
175  // Implementation based on {@link DynamicElementInfo#getEnclosingElementInfo()}
176  for (FieldInfo fi : ci.getDeclaredInstanceFields()) {
177  // Note name can be compiler dependent
178  // after the '$' is number representing depth of nesting
179  if (fi.getName().startsWith("this$")) {
180  return new StateValueElementInfoField(srvi, srvi.shouldExpandMembers(), newStateExpr, ei, fi);
181  }
182  }
183 
184  assert false; // Should be have outer class but not Field "this$"num not exists
185  throw new JPFInspectorNotInnerClassException(ci);
186  }
