114  public ExpressionBreakpointFieldAccess getBreakpointFieldAccess (BreakPointModes bpMode, FieldName fn) {
115  assert (bpMode == BreakPointModes.BP_MODE_FIELD_ACCESS) ||
116  (bpMode == BreakPointModes.BP_MODE_FIELD_ACCESS_READ)
117  || (bpMode == BreakPointModes.BP_MODE_FIELD_ACCESS_WRITE);
118  assert fn != null;
119  switch (bpMode) {
120  case BP_MODE_FIELD_ACCESS:
121  return new ExpressionBreakpointFieldAccess(AccessMode.ANY_ACCESS, fn);
122  case BP_MODE_FIELD_ACCESS_READ:
123  return new ExpressionBreakpointFieldAccess(AccessMode.READ, fn);
124  case BP_MODE_FIELD_ACCESS_WRITE:
125  return new ExpressionBreakpointFieldAccess(AccessMode.WRITE, fn);
126  default:
127  // Will never happen.
128  throw new RuntimeException("Unsupported BP mode.");
129  }
130  }

219  } catch (JPFInspectorGenericErrorException e) {
