450  private static String getSimpleName(String typeNameSignature) {
451  String fullName = demangleTypeName(typeNameSignature);
452 
453  int lastDot = fullName.lastIndexOf('.');
454  int lastDolar = fullName.lastIndexOf('$');
455 
456  int max = Math.max(lastDot, lastDolar);
457  if ((max > 0) && (max < (fullName.length() - 1))) {
458  fullName = fullName.substring(max + 1);
459  }
460  return fullName;
