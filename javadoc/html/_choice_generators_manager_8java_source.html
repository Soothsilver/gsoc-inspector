175  @Override
176  public void notifyChoiceGeneratorAdvance(ChoiceGenerator<?> cg, VM vm, InspectorState inspState) {
177  if (DEBUG) {
178  out.println(this.getClass().getSimpleName() + ".notifyChoiceGeneratorAdvance( cg=" + cg + ", inspState=" + inspState + ")");
179  }
180 
181  CGTypes cgType;
182  if (cg instanceof ThreadChoiceGenerator) {
183  cgType = CGTypes.CG_TYPE_SCHEDULING;
184  } else {
185  cgType = CGTypes.CG_TYPE_DATA;
186  }
187 
188  // Following default trace if exists
189  int dftChoice = forwardTrace.getDefaultChoice(cg); // Default forward trace choice
190  if (dftChoice != ChoiceGeneratorsInterface.NO_DEFAULT_CHOICE) {
191  cg.reset();
192  cg.advance(dftChoice);
193  if (DEBUG) {
194  out.println(this.getClass().getSimpleName() + ".notifyChoiceGeneratorAdvance - using default choice ... dftChoice=" + dftChoice + ", cg=" + cg);
195  }
196  }
197  // Remember the current choice
198  int currentChoice = cg.getProcessedNumberOfChoices() - 1;
199 
200  // Obtain all possible choices
201  cg.reset();
202  int totalChoices = cg.getTotalNumberOfChoices();
203  String[] choices = new String[totalChoices];
204  Object[] choiceObjects = new Object[totalChoices];
205  for (int i = 0; i < totalChoices; i++) {
206  cg.advance();
207  Object genChoice = cg.getNextChoice();
208  choiceObjects[i] = genChoice;
209  choices[i] = (genChoice != null ? genChoice.toString() : "null");
210  }
211 
212  // Restore state of the CG
213  cg.reset();
214  for (int i = 0; i <= currentChoice; i++) {
215  cg.advance();
216  }
217 
218  //out.println("Before: " + cg.getProcessedNumberOfChoices() + ", total: " + cg.getTotalNumberOfChoices());
219  if (cgType == CGTypes.CG_TYPE_SCHEDULING) {
220  // Skip disabled threads
221  for (int i = currentChoice; i < cg.getTotalNumberOfChoices(); i++) {
222  ThreadInfo threadInfo = (ThreadInfo)choiceObjects[i];
223  int targetId = threadInfo.getId();
224  if (isThreadDisabled(targetId)) {
225  out.println("Ignoring disabled thread " + targetId + ".");
226  cg.advance();
227  currentChoice++;
228  } else {
229  break;
230  }
231  }
232  //out.println("After: " + cg.getProcessedNumberOfChoices() + ", total: " + cg.getTotalNumberOfChoices());
233 
234  if (currentChoice >= choices.length) {
235  // The fact that the thread was disabled exhausted this choice generator.
236  vm.ignoreState();
237  }
238  }
239 
240  // Check whether send notification to client
241  boolean printChoice = cgNotifications[getIndexCGNotificationSpecification(cgType, CGMode.CG_MODE_PRINT)].isNotificationEnabled();
242  boolean askChoice = cgNotifications[getIndexCGNotificationSpecification(cgType, CGMode.CG_MODE_ASK)].isNotificationEnabled();
243  if (printChoice == false && askChoice == false) {
244  return; // No notification are selected
245  }
246 
247 
248 
249  int hashCode = cg.hashCode();
250 
251  // There was a to-do item here that said:
252  // "Gather default choice if exists"
253  // However, I have no idea what it means.
254  serverCallbacks.notifyChoiceGeneratorNewChoice(cgType, cg.getId(), hashCode, choices, currentChoice, ChoiceGeneratorsInterface.NO_DEFAULT_CHOICE);
255 
256  if (askChoice) {
257  serverCallbacks.specifyChoiceToUse(totalChoices - 1);
258 
259  waitForChoice = true;
260  resumeWasDueToStepping = false;
261  stopHolder.stopExecution(inspState);
262  waitForChoice = false;
263  if (resumeWasDueToStepping) {
264  serverCallbacks.genericInfo("You used a backstepping command while execution was broken due to a choice generator prompt." +
265  " Such use is not supported and will probably not work. It is recommended that you use breakpoints for breaking execution.");
266  }
267 
268  Object genChoice = cg.getNextChoice();
269  serverCallbacks.notifyUsedChoice(cgType, cg.getId(), hashCode, cg.getProcessedNumberOfChoices(), (genChoice != null ? genChoice.toString() : "null"));
270  }

278  @Override
279  public void selectChoice (int selectedChoice) throws JPFInspectorGenericErrorException {
280  assert stopHolder.isStopped();
281  VM vm = stopHolder.getVM();
282  assert vm != null;
283 
284  if (!waitForChoice) {
285  throw new JPFInspectorGenericErrorException("JPF has not been stopped by CG.Advanced call");
286  }
287 
288  ChoiceGenerator<?> cg = vm.getChoiceGenerator();
289 
290  if (cg == null) {
291  return;
292  }
293  // Processing default forward trace
294  int dftChoice = forwardTrace.getDefaultChoice(cg); // Default forward trace choice
295  if (selectedChoice == ChoiceGeneratorsInterface.USE_DEFAULT_CHOICE) {
296  if (dftChoice == ChoiceGeneratorsInterface.NO_DEFAULT_CHOICE) {
297  throw new JPFInspectorGenericErrorException("No forward trace to follow.");
298  }
299  //noinspection AssignmentToMethodParameter
300  selectedChoice = dftChoice;
301  }
302 
303  int totalChoices = cg.getTotalNumberOfChoices();
304 
305  if (selectedChoice < 0 || selectedChoice >= totalChoices) {
306  throw new JPFInspectorGenericErrorException("The index '" + selectedChoice + "' is not a valid choice.");
307  } else {
308  if (selectedChoice != dftChoice) {
309  // Choice different path then default forward one -> destroy DFT is exists
310  forwardTrace.destroyTrace(true);
311  }
312  cg.reset();
313  selectedChoice++; // Own index is zero based however (ThreadInfo CG) needs to advance one to get first thread
314 
315  cg.advance(selectedChoice);
316  }
317 
318  stopHolder.resumeExecution();
319  }
