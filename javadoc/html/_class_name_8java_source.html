76  /*
77  * Converts name of the class of file into regular expression patter.
78  * <br>
79  * The name may contain * to mark (eventually empty) number of arbitrary characters.
80  * Note that "." (and other special regexp characters) are escaped
81  * so that the * has special meaning.
82  * <br>
83  * Examples:
84  * <br> "MyEX*.java" is converted into MyEx.*\.java.
85  *
86  * @param name Name to convert
87  * @return regular expresison pattern
88  */
89  static public String convertName2RegExp(String name) {
90 
91  // Conversions ... we permit only * in regexp
92  // '\' -> "\\"
93  // '[' -> "\["
94  // ']' -> "\]"
95  // '}' -> "\{"
96  // '{' -> "\}"
97  // '(' -> "\("
98  // ')' -> "\)"
99  // '+' -> "\+"
100  // '^' -> "\^"
101  // '$' -> "\$"
102  // '?' -> "\?"
103  // ',' -> "\,"
104  // '&' -> "\&"
105  // '|' -> "\|"
106  name = name.replace( "\\", "\\\\");
107  name = name.replace( "[", "\\[");
108  name = name.replace( "]", "\\]");
109  name = name.replace( "}", "\\}");
110  name = name.replace( "{", "\\{");
111  name = name.replace( "(", "\\(");
112  name = name.replace( ")", "\\)");
113  name = name.replace( "+", "\\+");
114  name = name.replace( "^", "\\^");
115  name = name.replace( "$", "\\$");
116  name = name.replace( "?", "\\?");
117  name = name.replace( ",", "\\,");
118  name = name.replace( "&", "\\&");
119  name = name.replace( "|", "\\|");
120 
121  // '.' -> "\." not any char
122  // '*' -> ".*"
123  name = name.replace( ".", "\\.");
124  name = name.replace( "*", ".*");
125  return "^" + name + "\\z";
126  }
