77  public InspectorPrimaryConsolePanel() {
78  super("JPF Inspector", null, "JPF Inspector command console");
79  Config config = ShellManager.getManager().getConfig();
80  String target = config.getTarget();
81 
82  // Create a terminal with specified theme
83  if (config.get("jpf-inspector.theme").equals("white_on_black")) {
84  terminal = SwingTerminal.getSwingTerminalBlack();
85  log.info("JPF Inspector theme: white-on-black.");
86  } else if (config.get("jpf-inspector.theme").equals("black_on_white")) {
87  terminal = SwingTerminal.getSwingTerminalWhite();
88  log.info("JPF Inspector theme: black-on-white.");
89  }
90  else {
91  terminal = SwingTerminal.getSwingTerminalWhite();
92  log.warning("Theme '" + config.get("jpf-inspector.theme") + "' not recognized. Defaulting to black-on-white.");
93  }
94 
95  // Make sure the terminal is selected whenever we are on this panel
96  addComponentListener(new PanelComponentListener());
97 
98  // Prepare communication streams
99  consolePrintStream = terminal.getUserTextPrintStream();
100  PrintStream startingInfoStream = terminal.getSimplePrintStream();
101  console = terminal.getConsoleReader();
102 
103  // Initialize server part
104  inspectorClient = JPFInspectorFacade.getInspectorClient(target, consolePrintStream);
105 
106  // Register InspectorPrimaryConsolePanel as a Verify command listener
107  ShellManager.getManager().addCommandListener(VerifyCommand.class, this);
108 
109  // Set up the layout
110  this.setLayout(new BorderLayout());
111  add(terminal.getScrollPanel());
112 
113  // Say hello to the user
114  startingInfoStream.println("This is the JPF Inspector console for debugging the target \"" + target + "\".");
115  startingInfoStream.println("Type \"hello\" to test if the Inspector is working or \"help\" to get a list of commands.");
116  startingInfoStream.println();
117 
118  // Set up quick commands
119  setUpQuickCommands();
120 
121  // Start up the command execution thread
122  Thread t = new CommandProcessingThread();
123  t.start();
124  }

188  private class CommandProcessingThread extends Thread {
189  public CommandProcessingThread () {
190  super(CommandProcessingThread.class.getSimpleName());
191  }

193  @Override
194  public void run () {
195  //noinspection InfiniteLoopStatement
196  while (true) {
197  try {
198  String s = console.readLine(Constants.PROMPT);
199  if (s.trim().isEmpty()) {
200  consolePrintStream.println("\n" + Constants.PROMPT);
201  } else {
202  consolePrintStream.println();
203  getInspectorClient().executeCommand(s.trim(), ExecutionContext.FROM_SWING_TERMINAL);
204  }
205  } catch (IOException e) {
206  consolePrintStream.println("ERR: Error while reading a command.");
207  }
208  } // end of while
209  }
210  }
