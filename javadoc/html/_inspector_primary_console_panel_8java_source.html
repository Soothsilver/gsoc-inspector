79  public InspectorPrimaryConsolePanel() {
80  super("JPF Inspector", null, "JPF Inspector command console");
81  Config config = ShellManager.getManager().getConfig();
82  String target = config.getTarget();
83 
84  // Create a terminal with specified theme
85  if (config.get("jpf-inspector.theme").equals("white_on_black")) {
86  terminal = SwingTerminal.getSwingTerminalBlack();
87  log.info("JPF Inspector theme: white-on-black.");
88  } else if (config.get("jpf-inspector.theme").equals("black_on_white")) {
89  terminal = SwingTerminal.getSwingTerminalWhite();
90  log.info("JPF Inspector theme: black-on-white.");
91  }
92  else {
93  terminal = SwingTerminal.getSwingTerminalWhite();
94  log.warning("Theme '" + config.get("jpf-inspector.theme") + "' not recognized. Defaulting to black-on-white.");
95  }
96 
97  // Make sure the terminal is selected whenever we are on this panel
98  addComponentListener(new PanelComponentListener());
99 
100  // Prepare communication streams
101  consolePrintStream = terminal.getUserTextPrintStream();
102  PrintStream startingInfoStream = terminal.getSimplePrintStream();
103  console = terminal.getConsoleReader();
104 
105  // Initialize server part
106  inspector = JPFInspectorFacade.getInspectorClient(target, consolePrintStream);
107 
108  // Register InspectorPrimaryConsolePanel as a Verify command listener
109  ShellManager.getManager().addCommandListener(VerifyCommand.class, this);
110 
111  // Set up the layout
112  this.setLayout(new BorderLayout());
113  add(terminal.getScrollPanel());
114 
115  // Say hello to the user
116  startingInfoStream.println("This is the JPF Inspector console for debugging the target \"" + target + "\".");
117  startingInfoStream.println("Type \"hello\" to test if the Inspector is working or \"help\" to get a list of commands.");
118  startingInfoStream.println();
119 
120  // Set up quick commands
121  setUpQuickCommands();
122 
123  // Start up the command execution thread
124  Thread t = new CommandProcessingThread();
125  t.start();
126  }

159  private class CommandProcessingThread extends Thread {
160  public CommandProcessingThread () {
161  super(CommandProcessingThread.class.getSimpleName());
162  }

164  @Override
165  public void run () {
166  //noinspection InfiniteLoopStatement
167  while (true) {
168  try {
169  String s = console.readLine(Constants.PROMPT);
170  if (s.trim().isEmpty()) {
171  consolePrintStream.println("\n" + Constants.PROMPT);
172  } else {
173  consolePrintStream.println();
174  inspector.executeCommand(s.trim(), ExecutionContext.FROM_SWING_TERMINAL);
175  }
176  } catch (IOException e) {
177  consolePrintStream.println("ERR: Error while reading a command.");
178  }
179  } // end of while
180  }
181  }
