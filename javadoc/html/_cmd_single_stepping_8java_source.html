92  @Override
93  public void execute(JPFInspectorClient client, JPFInspectorBackEndInterface inspector, PrintStream outStream) {
94  assert inspector != null;
95  if (stepCount <= 0) {
96  outStream.println("The number of repetitions must be 1 or greater.");
97  return;
98  }
99 
100  if (isBackwardsTransitionStep()) {
101  try {
102  inspector.backstepTransition(stepType, stepCount);
103  } catch (JPFInspectorException e) {
104  outStream.println(e.getMessage());
105  client.recordComment(e.getMessage());
106  }
107  } else {
108  for (int i = 0; i < stepCount; i++) {
109  try {
110  if (forward) {
111  inspector.forwardStep(stepType);
112  } else {
113  if (fieldNameExpression != null) {
114  inspector.backFieldAccessStep(fieldNameExpression);
115  } else {
116  inspector.backstep(stepType);
117  }
118  }
119  } catch (JPFInspectorException e) {
120  outStream.println(e.getMessage());
121  client.recordComment(e.getMessage());
122  }
123  boolean isFinalStep = (i == (stepCount - 1));
124  if (!isFinalStep) {
125  inspector.waitUntilStopped();
126  }
127  }
128  }
129  if (InspectorConfiguration.getInstance().shouldWaitAfterRun()) {
130  inspector.waitUntilStopped();
131  }
132  }
