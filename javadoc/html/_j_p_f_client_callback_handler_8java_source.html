67  private static boolean checkIfShowStateChangeNofitication (CB_METHODS prevCB, InspectorStates newState) {
68  if (prevCB == null) {
69  return true; // First CB
70  }
71 
72  if (InspectorStates.JPF_TERMINATING.equals(newState) || InspectorStates.JPF_STARTED.equals(newState)) {
73  return true; // Always show JPF started and terminating
74  }
75 
76  if (CB_METHODS.CB_BREAKPOINT_HIT.equals(prevCB) || CB_METHODS.CB_CG_CHOICE_TO_USE.equals(prevCB)) {
77  return false; // Don't show stop notification after breakpoint hit
78  }
79 
80  //noinspection RedundantIfStatement
81  if (InspectorStates.JPF_RUNNING.equals(newState) && CB_METHODS.CB_STATE_CHANGE.equals(prevCB)) {
82  // Suppose previous SB was JPF Stopped ad next command started JPF execution (run, (back_)step_over, )
83  // --> hide obvious JPF started notification
84  return false;
85  }
86 
87  return true; // Fail safe
88  }

90  @Override
91  public void notifyStateChange (InspectorStates newState, String details) {
92  // Check if notification can be hidden
93  if (checkIfShowStateChangeNofitication(prevCB, newState)) {
94 
95  if (newState == InspectorStates.JPF_STARTED) {
96  out.println("INFO: JPF created and connected, SuT is started");
97  } else if (newState == InspectorStates.JPF_RUNNING) {
98  out.println("INFO: SuT is running");
99  } else if (newState == InspectorStates.JPF_STOPPED) {
100  out.println("INFO: SuT is stopped");
101  if (details != null && !details.trim().isEmpty()) {
102  out.println("\t" + details);
103  }
104  } else if (newState == InspectorStates.JPF_TERMINATING) {
105  out.println("INFO: JPF Terminating");
106  } else {
107  throw new RuntimeException("Unknown " + InspectorStates.class.getName() + " enum entry " + newState);
108  }
109  }
110  prevCB = CB_METHODS.CB_STATE_CHANGE;
111  if (newState == InspectorStates.JPF_TERMINATING) {
112  prevCB = null;
113  }
114 
115  }

117  @Override
118  public void notifyBreakpointHit (BreakPointStatus bp) {
119  assert bp != null;
120 
121  if (bp instanceof AssertStatus) {
122  AssertStatus as = (AssertStatus) bp;
123 
124  String message = "Assertion violated.";

126  message = "Logging assertion violated";
127  }
128  out.println("INFO: " + message + " [assert " + as.getNormalizedPosition() + " " + as.getNormalizedCondition() + "]");
129 
130  } else {
131  // Standard breakpoint
132  String message = "Breakpoint hit:";
133  if (bp.getState() == BreakPointStates.BP_STATE_LOGGING) {
134  message = "Logging breakpoint hit:";
135  }
136  out.println("INFO: " + message + " [" + CmdBreakpointShow.breakpointToString(bp) + "]");
137  String details = bp.getDetails();
138  if (details != null && !details.trim().isEmpty()) {
139  out.println("\t" + details);
140  }
141  }
142  prevCB = CB_METHODS.CB_BREAKPOINT_HIT;
143  }
