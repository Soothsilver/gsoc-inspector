100  private Step backtracker (int howManyMethodsToLeave) {
101  int callStackDepth = howManyMethodsToLeave;
102  // Simulation of the depth of the calls stack above (how many methods have been called by)
103  // the method in the end of the path.
104 
105  // We always want to perform at least one backstep.
106  Step step = stepThreadBacktracker.backstep();
107 
108  while (step != null) {
109  if (returnChecker.isReturnStep(step)) {
110  callStackDepth++;
111  }
112  if (throwChecker.isThrowStep(step)) {
113  throw new UnsupportedOperationException(
114  "For implementation reasons of the Inspector, it is not possible to backtrack through the throwing of an exception.");
115  }
116  if (callChecker.isCallStep(step)) {
117  callStackDepth--;
118  }
119  if (DEBUG) {
120  System.out.print(" |");
121  for (int i = 0; i < callStackDepth; i++) {
122  System.out.print(" |");
123  }
124  if (returnChecker.isReturnStep(step)) {
125  System.out.print("-");
126  } else if (callChecker.isCallStep(step)) {
127  // System.out.print(" ");
128  } else {
129  System.out.print(" ");
130  }
131 
132  System.out.println(
133  " inst=" + step.getInstruction() + ", location=" + step.getInstruction().getFilePos() + ", source="
134  + step.getInstruction().getSourceLine());
135  }
136 
137  // Call stack depth indicates how many more methods must we leave before we want to stop.
138  // If it's zero, it means that we already want to stop.
139  // For now, it's impossible for call stack depth to be less than zero.
140  if (callStackDepth <= 0) {
141  return step;
142  } else {
143  step = stepThreadBacktracker.backstep();
144  }
145  }
146 
147  // This means the backstep failed and there is nothing to backtrack to, we are in the first instruction
148  // of the thread, we failed.
149  return null;
150  }

179  public static int howManySameInstructionStepsUpToStep(Transition transition, Step step) {
180  assert (transition != null);
181  assert (step != null);
182 
183  final Instruction stInst = step.getInstruction();
184 
185  int count = 0;
186  Step trStep = transition.getStep(0);
187  while (trStep != null) {
188 
189  // Same instruction
190  if (stInst.equals(trStep.getInstruction())) {
191  count++;
192  }
193 
194  // Are all instruction before processed?
195  if (step.equals(trStep)) {
196  break;
197  }
198  trStep = trStep.getNext();
199  } // End while
200 
201  assert count >= 1;
202  return count;
203  }

205  public BackwardBreakpointCreator backtrackIn(InstructionPosition currentLocation) {
206  // We must first leave the current line.
207  Step step = backtracker(0); // We must make at least one step.
208  while (step != null && currentLocation.hitPosition(step.getInstruction())) {
209  step = backtracker(0);
210  }
211  if (step == null) {
212  // Beginning of thread.
213  return null;
214  }
215  // Now we're on the previous line, either in the same method or in the caller.
216 
217  InstructionPosition previousLinePosition = InstructionPositionImpl.getInstructionPosition(step.getInstruction());
218  // Be careful - we need to remember previous step and transition2backreack
219  Step previousStep = null;
220  Transition previousTransition = null;
221  int previousTransitionsToBacktrack = 0;
222  ReturnInstructionChecker returnInstructionChecker = new ReturnInstructionChecker();
223  while (step != null && previousLinePosition.hitPosition(step.getInstruction())) {
224  previousStep = step;
225  previousTransition = getCurrentTransition();
226  previousTransitionsToBacktrack = getBacktrackedTransitionCount();
227  step = stepThreadBacktracker.backstep();
228  if (returnInstructionChecker.isReturnStep(step)) {
229  return new BackwardBreakpointCreator(getCurrentTransition(), step, getBacktrackedTransitionCount());
230  }
231  }
232  if (step == null) {
233  return null;
234  }
235  return new BackwardBreakpointCreator(previousTransition, previousStep, previousTransitionsToBacktrack);
236  }
