152  public StateNodeInterface getExpressionFromStackFrame(StateStackFrame parent) throws JPFInspectorException {
153  assert (parent != null);
154 
155  StateReadableValue resultingValue = null;
156 
157  // Local variable or parameter
158  if (parent.namedSlotIndex(varName) != StateValueStackSlot.INVALID_SLOT_INDEX) {
159  resultingValue = StateValueStackSlot.createNamedSlotValue(parent, varName);
160  }
161 
162  ClassInfo ci = parent.getClassInfo();
163  // For some synthetic bytecode instructions ClassInfo can be null
164  // ci == null should not occur for standard user loaded bytecode
165  if (ci != null) {
166  // Field or static field
167  if (resultingValue == null) {
168  if (parent.isStaticMethod()) {
169  FieldInfo fi = ci.getStaticField(varName);
170  if (fi != null) {
171  resultingValue = StateValueElementInfoField.createStaticNamedField(parent, varName);
172  }
173  } else {
174  FieldInfo fi = StateValueElementInfoField.fieldNameJavaBasedLookup(ci, varName);
175  if (fi != null) {
176  // Correct not a static
177  StateValueStackSlot svsf = StateValueStackSlot.createHiddenThisSlotValue(parent);
178  if (fi.isStatic()) {
179  resultingValue = StateValueElementInfoField.createStaticNamedField(svsf, varName);
180  } else {
181  resultingValue = StateValueElementInfoField.createInstanceNamedField(svsf, varName);
182  }
183  }
184  }
185  }
186 
187  // Predecessor class Name
188  if (resultingValue == null) {
189  ClassInfo ciPred = StateElementInfo.isPredecessorTypeName(ci, varName);
190 
191  if (ciPred != null) {
192  if (parent.isStaticMethod()) {
193  resultingValue = StateElementInfo.createStaticClass(parent, ciPred);
194  } else {
195  StateValueStackSlot svsf = StateValueStackSlot.createHiddenThisSlotValue(parent);
196  resultingValue = svsf.createPredecessorClass(ciPred);
197  }
198  } else {
199  // Not a predecessor class (any other class)
200  // We are using the final, unqualified name, only. Really, this is undocumented and will stay that way for now,
201  // because exploring static area is not something we are well equipped to do at this time anyway.
202  StaticElementInfo sei = null;
203  Statics statics = ClassLoaderInfo.getCurrentClassLoader().getStatics();
204  for (int i = 0; i < statics.size(); i++) {
205  StaticElementInfo aStaticClass = statics.get(i);
206  if ((aStaticClass.getClassInfo().getName().endsWith("." + varName))) {
207  sei = aStaticClass;
208  break;
209  }
210  }
211  if (sei != null) {
212  // Class with given name exists
213  resultingValue = StateElementInfo.createStaticClass(parent, sei.getClassInfo());
214  }
215  /*
216  Code before migration:
217  JPF jpf = ssf.getInspector().getJPF();
218  StaticArea sa = jpf.getVM().getStaticArea();
219  StaticElementInfo sei = sa.get(varName);
220  if (sei != null) {
221  // Class with given name exists
222  srvi = StateElementInfo.createStaticClass(ssf, sei.getClassInfo());
223  }
224  */
225  }
226  }
227  }
228 
229  if (resultingValue == null) {
230  // Double hacks.
231  resultingValue = tryApplyDoubleHacks(parent);
232  }
233 
234  if (resultingValue == null) {
235  throw new JPFInspectorInvalidNameException(varName);
236  }
237 
238  ExpressionStateValue child = getChild();
239  if (child == null) {
240  return resultingValue;
241  } else {
242  return child.toHierarchy2(resultingValue);
243  }
244  }
