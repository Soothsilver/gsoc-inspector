141  public StateNodeInterface getResultExpression (StateStackFrame ssf) throws JPFInspectorException {
142  assert (ssf != null);
143 
144  StateReadableValueInterface srvi = null;
145 
146  // Local variable or parameter
147  if (ssf.namedSlotIndex(varName) != StateValueStackSlot.INVALID_SLOT_INDEX) {
148  srvi = StateValueStackSlot.createNamedSlotValue(ssf, varName);
149  }
150 
151  ClassInfo ci = ssf.getClassInfo();
152  // For some synthetic bytecode instructions can be ClassInfo null
153  // ci == null should not occur for standard user loaded bytecode
154  if (ci != null) {
155 
156  // Field or static field
157  if (srvi == null) {
158  if (ssf.isStaticMethod()) {
159  FieldInfo fi = ci.getStaticField(varName);
160  if (fi != null) {
161  srvi = StateValueElementInfoField.createStaticNamedField(ssf, varName);
162  }
163  } else {
164  FieldInfo fi = StateValueElementInfoField.fieldNameJavaBasedLookup(ci, varName);
165  if (fi != null) {
166  // Correct not a static
167  StateValueStackSlot svsf = StateValueStackSlot.createHiddenThisSlotValue(ssf);
168  if (fi.isStatic()) {
169  srvi = StateValueElementInfoField.createStaticNamedField(svsf, varName);
170  } else {
171  srvi = StateValueElementInfoField.createInstanceNamedField(svsf, varName);
172  }
173  }
174  }
175  }
176 
177  // Predecessor class Name
178  if (srvi == null) {
179  ClassInfo ciPred = StateElementInfo.isPredecessorTypeName(ci, varName);
180 
181  if (ciPred != null) {
182  if (ssf.isStaticMethod()) {
183  srvi = StateElementInfo.createStaticClass(ssf, ciPred);
184  } else {
185  StateValueStackSlot svsf = StateValueStackSlot.createHiddenThisSlotValue(ssf);
186  srvi = svsf.createPredecessorClass(ciPred);
187  }
188  } else {
189  // Not a predecessor class (any other class)
190  StaticElementInfo sei = MigrationUtilities.getStaticElementInfoFromName(varName);
191 
192  if (sei != null) {
193  // Class with given name exists
194  srvi = StateElementInfo.createStaticClass(ssf, sei.getClassInfo());
195  }
196  /*
197  Code before migration:
198  JPF jpf = ssf.getInspector().getJPF();
199  StaticArea sa = jpf.getVM().getStaticArea();
200  StaticElementInfo sei = sa.get(varName);
201  if (sei != null) {
202  // Class with given name exists
203  srvi = StateElementInfo.createStaticClass(ssf, sei.getClassInfo());
204  }
205  */
206  }
207  }
208  }
209 
210  if (srvi == null) {
211  // Double hacks.
212  srvi = tryApplyDoubleHacks(ssf);
213  }
214 
215  if (srvi == null) {
216  throw new JPFInspectorInvalidNameException(varName);
217  }
218 
219  // Childs
220  ExpressionStateValue child = getChild();
221  if (child == null) {
222  return srvi;
223  } else {
224  return child.getResultExpression(srvi);
225  }
