95  @Override
96  public boolean evaluateExpression (InspectorState state) {
97  if (state.getListenerMethod() == ListenerMethod.LM_EXECUTE_INSTRUCTION) {
98  VM vm = state.getVM();
99  assert vm != null;
100  ThreadInfo ti = vm.getCurrentThread();
101  assert ti.getId() == threadNum;
102  if (ti.getId() == threadNum) {
103  Instruction executedInstr = vm.getInstruction();
104  Debugging.getLogger().info("Backtracking: Forward stepping: Testing against the same thread (" + executedInstr.getMnemonic() + "," + executedInstr.getFileLocation() + ")");
105  if (instruction.equals(executedInstr)) {
106  count++;
107  return count == hitCount;
108  }
109  }
110  }
111  return false;
112  }

124  static ExpressionBreakpointInstruction getBPonPreviousInstruction (Transition tr) {
125  assert tr != null;
126  int instrIndex = tr.getStepCount() - 1;
127  if (instrIndex < 1) {
128  return null;
129  }
130  Step penultimateStep = tr.getStep(instrIndex);
131  Instruction penultimateInstr = penultimateStep.getInstruction();
132 
133  int sameInstrCount = 1; // One for the final instruction where we hit the BP
134  // Count how many time the instruction takes place in the transition (due to cycle repetitions etc.)
135  for (Step currentStep = tr.getStep(0); !currentStep.equals(penultimateStep); currentStep = currentStep.getNext()) {
136  if (penultimateInstr.equals(currentStep.getInstruction())) {
137  sameInstrCount++;
138  }
139  }
140 
141  return new ExpressionBreakpointInstruction(tr.getThreadIndex(), penultimateInstr, sameInstrCount);
142  }

170  assert (methodSpec != null);
171  ClassName classSpec = methodSpec.getClassNameClass();
172 
173 
174  // The following block used to be the simple assignment
175  // "ClassInfo[] cis = ClassLoader.getLoadedClass();"
176  // However, that method is no longer available in JPF8.
177  ArrayList<ClassInfo> loadedClasses = new ArrayList<>();
178  for (ClassInfo loadedClass : ClassLoaderInfo.getCurrentClassLoader()) {
179  loadedClasses.add(loadedClass);
180  }
181  ClassInfo[] cis = new ClassInfo[loadedClasses.size()];
182  cis = loadedClasses.toArray(cis);
183  // End of block.
184 
185  for (ClassInfo ci : cis) {
186  assert (ci != null);
187  if (classSpec.isSameClass(ci)) {
188  MethodInfo[] mis = ci.getDeclaredMethodInfos();
189  for (MethodInfo mi : mis) {
190  if (methodSpec.isSameMethod(mi)) {
191  return mi.getInstruction(instIdx);
192  }
193  }
194  }
195  }
196  // No such method exists
197  return null;
