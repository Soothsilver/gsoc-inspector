60  public StateNodeInterface getResultExpression (StateThreadInfo thread) throws JPFInspectorException {
61  assert thread != null;
62 
63  StateStackFrame thisStackFrame = new StateStackFrame(thread, stackFrameNum);
64 
65  ExpressionStateValue child = getChild();
66  if (child == null) {
67  return thisStackFrame;
68  } else {
69  // #stackSlot
70  if (child instanceof ExpressionStateStackFrameSlot) {
71  ExpressionStateStackFrameSlot childSfs = (ExpressionStateStackFrameSlot) child;
72  return childSfs.getExpressionFromStackFrame(thisStackFrame);
73  // #this (also comes from #super, #outerClass etc...)
74  } else if (child instanceof ExpressionStateValueThis) {
75  ExpressionStateValueThis childThis = (ExpressionStateValueThis) child;
76  return childThis.getExpressionFromStackFrame(thisStackFrame);
77  // a name
78  } else if (child instanceof ExpressionStateValueName) {
79  ExpressionStateValueName childName = (ExpressionStateValueName) child;
80  return childName.getExpressionFromStackFrame(thisStackFrame);
81  // #static
82  } else if (child instanceof ExpressionStateValueStatic) {
83  ExpressionStateValueStatic childStatic = (ExpressionStateValueStatic) child;
84  return childStatic.getExpressionFromStackFrame(thisStackFrame);
85  } else {
86  throw new RuntimeException("Invalid child type.");
87  }
88  }
