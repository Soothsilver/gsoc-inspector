70  public Step backstep() {
71  Debugging.getLogger().info("Determining target instruction: Backstep: Commencing.");
72  // If we are the start of a transition, and before we begin, we must request a new transition.
73  if (currentTransitionStepIndex <= 0) {
74  if (!requestPreviousTransition()) {

98  private boolean requestPreviousTransition() {
99  assert (currentTransitionStepIndex == 0 ||
100  currentTransitionStepIndex == BACKSTEP_NOT_YET_CALLED); // Otherwise no reason to call this method
101 
102  Step previousStep = null;
103  Transition previousTransition = ttb.getCurrentTransition(); // will be previousTransition
104  if (previousTransition != null) {
105  previousStep = previousTransition.getStep(0);
106  }
107 
108  // Move to older transition
109  Transition currentTransition = ttb.backtrackToPreviousTransition();
110  if (currentTransition == null) {
111  // We have processed whole trace for given thread
112  currentTransitionSteps = null;
113  currentTransitionStepIndex = 0;
114  return false;
115  }
116 
117  currentTransitionSteps = transition2StepArray(currentTransition);
118  currentTransitionStepIndex = currentTransitionSteps.length;
119  assert currentTransitionStepIndex > 0;
120  if (currentTransitionStepIndex > 0) {
121  // Check if the last step in current transition is not the bottom half
122  // of the first step in previous (consequent) transition
123  if (previousStep != null) {
124  Instruction instCT = currentTransitionSteps[currentTransitionStepIndex - 1].getInstruction();
125  assert (instCT != null);
126 
127  if (instCT.equals(previousStep.getInstruction())) {
128  currentTransitionStepIndex--; // Bottom half of the instruction should be ignored
129  }
130  }
131  }
132  return true;
133  }
