163  @Override
164  public synchronized void bindWithJPF (JPF jpf) throws JPFInspectorGenericErrorException {
165  if (DEBUG) {
166  getDebugPrintStream().println(" " + JPFInspector.class.getSimpleName() + ".boundWithJPF(jpf=" + jpf + " )");
167  }
168  if (jpf == null) {
169  return;
170  }
171 
172  if ((this.jpf != null) && (listener.getFinished() == false)) {
173  stopHolder.terminating();
174  // Currently there is running older instance ... we should terminate it
175  // TODO: Is not better to force previous instance to stop.
176  try {
177  wait();
178  } catch (InterruptedException e) {
179  throw new JPFInspectorGenericErrorException("Previous instance of the JPF is running");
180  }
181  }
182 
183  // Modify JPF Configuration
184  jpf.getVM().recordSteps(true);
185  Config jpfCfg = jpf.getConfig();
186 
187  // Previously, there was a complex block of code here that I did not understand.
188  // It felt like it was trying to force a multiple-error search regardless of the "search.multiple_error"
189  // setting, but it also referenced a change "to be propagated to the trunk" which never made it to the trunk.
190  // I changed it to respect the property, hopefully that won't break it.
191  boolean originalSearchMultipleErrors = jpfCfg.getBoolean("search.multiple_errors");
192 
193  this.jpf = jpf;
194  this.jpfHasLaunched = true;
195  notifyAll(); // JPF is now connected and the "run" command should be notified.
196  // This also requires synchronization.
197 
198  listener = new InspectorListener(this, commandsManager, breakpointHandler, choiceGeneratorsManager,
199  defaultForwardTraceManager, originalSearchMultipleErrors);
200  jpf.addListener(listener);

202  // Initialize instance of the JPF

205  // There is no way how to swap search objects

207  } else {

209  searchInspector.setInspector(this);
