145  public void stopExecution (InspectorState inspState) {
146  assert inspState != null;
147 
148  boolean terminate; // Local synchronized version of this.terminating variable
149  accessLock.lock();
150  terminate = terminating; // Use single value even if we leave this critical section
151 
152  try {
153  if (!terminate) {
154  this.inspState = inspState;
155  stopped = true;
156 
157  serverCallbacks.notifyStateChange(InspectorStatusChange.JPF_STOPPED, getLocationDetails(inspState));
158  //hasJpfStoppedAtLeastOnce = true;
159 
160  accessLockConditionVariable.signalAll(); // Notify all threads waiting for JPF to be stopped (they are woken up after the wait)
161  accessLockConditionVariable.await();
162 
163  while (hostageHolders > 0) {
164  jpfResumePreventionConditionVariable.await();
165  }
166 
167  stopped = false;
168

171  this.inspState = null;

175  }

177  stopped = false;
178  if (DEBUG) {
179  inspector.getDebugPrintStream().println(this.getClass().getSimpleName() + ": INTERRUPTED.");

181  }

186  if (terminate) {
187  notifyClientTerminating();
188  } else {
189  serverCallbacks.notifyStateChange(InspectorStatusChange.JPF_RUNNING, getLocationDetails(inspState));
