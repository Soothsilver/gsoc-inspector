169  public boolean evaluateBreakpoint (InspectorState state) {
170  boolean bpHitted = false;
171 
172  // Hit condition
173  if (bpExpression != null) {
174  try {
175  bpHitted = bpExpression.evaluateExpression(state);
176  } catch (JPFInspectorException e) {
177  // Ignore the exception Breakpoint cannot be evaluated --> thus does not hold
178  }
179  }
180 
181  MethodInfo topFrameMethodInfo = state.getVM().getCurrentThread().getTopFrameMethodInfo();
182  if (bpHitted && topFrameMethodInfo != null) {
183  if (log.isLoggable(Level.FINE)) {
184  log.fine("Now checking within class: " + topFrameMethodInfo.getClassName());
185  }
186  if (InspectorConfiguration.getInstance().isClassIgnored(topFrameMethodInfo.getClassName())) {
187  bpHitted = false;
188  }
189  }
190 
191  if (bpHitted) {
192  bpHitCounter++;
193  bpHitCounterTotal++;
194  }
195 
196  boolean bpShouldExecuteAction = bpHitted;
197  if (lowerBound != null && bpHitCounter < lowerBound) {
198  bpShouldExecuteAction = false;
199  }
200 
201  if (upperBound != null && bpHitCounter > upperBound) {
202  bpShouldExecuteAction = false;
203  }
204 
205 
206 
207  if (bpShouldExecuteAction && isUserBreakpoint()) {
208  if (bpState == BreakpointState.BP_STATE_DISABLED) {
209  // No action expected
210  } else if (bpState == BreakpointState.BP_STATE_LOGGING) {
211  callbacks.notifyBreakpointHit(getBreakpointStatus(state));
212  } else if (bpState == BreakpointState.BP_STATE_ENABLED) {
213  callbacks.notifyBreakpointHit(getBreakpointStatus(state));
214  } else {
215  throw new RuntimeException("Unknown " + BreakpointState.class.getSimpleName() + " entry " + bpState);
216  }
217  }
218 
219  return bpShouldExecuteAction && bpState == BreakpointState.BP_STATE_ENABLED;
220 
221  }
