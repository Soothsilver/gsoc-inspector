182  public boolean evaluateBreakpoint (InspectorState state) {
183  boolean bpHitted = false;
184 
185  // Hit condition
186  if (bpExpression != null) {
187  try {
188  bpHitted = bpExpression.evaluateExpression(state);
189  } catch (JPFInspectorException e) {
190  // Ignore the exception Breakpoint cannot be evaluated --> thus does not hold
191  }
192  }
193 
194  MethodInfo topFrameMethodInfo = state.getVM().getCurrentThread().getTopFrameMethodInfo();
195  if (bpHitted && topFrameMethodInfo != null) {
196  if (log.isLoggable(Level.FINE)) {
197  log.fine("Now checking within class: " + topFrameMethodInfo.getClassName());
198  }
199  if (this.userBP && InspectorConfiguration.getInstance().isClassIgnored(topFrameMethodInfo.getClassName())) {
200  bpHitted = false;
201  }
202  }
203 
204 
205  if (bpHitted) {
206  bpHitCounter++;
207  bpHitCounterTotal++;
208  }
209 
210  boolean bpShouldExecuteAction = bpHitted;
211  if (lowerBound != null && bpHitCounter < lowerBound) {
212  bpShouldExecuteAction = false;
213  }
214 
215  if (upperBound != null && bpHitCounter > upperBound) {
216  bpShouldExecuteAction = false;
217  }
218 
219 
220 
221  if (bpShouldExecuteAction && isUserBreakpoint()) {
222  if (breakpointState == BreakpointState.DISABLED) {
223  // No action expected
224  } else if (breakpointState == BreakpointState.LOGGING) {
225  callbacks.notifyBreakpointHit(getBreakpointStatus(state));
226  } else if (breakpointState == BreakpointState.ENABLED) {
227  callbacks.notifyBreakpointHit(getBreakpointStatus(state));
228  } else {
229  throw new RuntimeException("Unknown " + BreakpointState.class.getSimpleName() + " entry " + breakpointState);
230  }
231  }
232 
233  return bpShouldExecuteAction && breakpointState == BreakpointState.ENABLED;
234 
235  }
