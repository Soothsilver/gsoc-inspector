177  public boolean evaluateBreakpoint (InspectorState state) {
178  boolean bpHitted = false;
179 
180  // Hit condition
181  if (bpExpression != null) {
182  try {
183  bpHitted = bpExpression.evaluateExpression(state);
184  } catch (JPFInspectorException e) {
185  // Ignore the exception Breakpoint cannot be evaluated --> thus does not hold
186  }
187  }
188 
189  MethodInfo topFrameMethodInfo = state.getVM().getCurrentThread().getTopFrameMethodInfo();
190  if (bpHitted && topFrameMethodInfo != null) {
191  if (log.isLoggable(Level.FINE)) {
192  log.fine("Now checking within class: " + topFrameMethodInfo.getClassName());
193  }
194  if (InspectorConfiguration.getInstance().isClassIgnored(topFrameMethodInfo.getClassName())) {
195  bpHitted = false;
196  }
197  }
198 
199  if (bpHitted) {
200  bpHitCounter++;
201  bpHitCounterTotal++;
202  }
203 
204  boolean bpShouldExecuteAction = bpHitted;
205  if (lowerBound != null && bpHitCounter < lowerBound) {
206  bpShouldExecuteAction = false;
207  }
208 
209  if (upperBound != null && bpHitCounter > upperBound) {
210  bpShouldExecuteAction = false;
211  }
212 
213 
214 
215  if (bpShouldExecuteAction && isUserBreakpoint()) {
216  if (breakpointState == BreakpointState.DISABLED) {
217  // No action expected
218  } else if (breakpointState == BreakpointState.LOGGING) {
219  callbacks.notifyBreakpointHit(getBreakpointStatus(state));
220  } else if (breakpointState == BreakpointState.ENABLED) {
221  callbacks.notifyBreakpointHit(getBreakpointStatus(state));
222  } else {
223  throw new RuntimeException("Unknown " + BreakpointState.class.getSimpleName() + " entry " + breakpointState);
224  }
225  }
226 
227  return bpShouldExecuteAction && breakpointState == BreakpointState.ENABLED;
228 
229  }
