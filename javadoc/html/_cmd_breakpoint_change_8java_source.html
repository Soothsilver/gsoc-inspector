63  if (!rc) {
64  outStream.println("Breakpoint with ID " + breakpointIdInteger + " does not exist.");
65  } else {
66  String newStatus = "enabled";

68  newStatus = "a logging breakpoint";
69  }
70  if (newState == BreakpointState.DISABLED) {

74  }
75  } catch (NumberFormatException e) {
