71  private static void printCommands(PrintStream outStream) {
72  LinkedHashMap<String, List<CommandHelpInfo>> categories = new LinkedHashMap<>();
73  // Linked hash map, because we want to preserve our order of categories
74 
75  addBuiltInCommands(categories);
76 
77  addAliases(categories);
78 
79  addCustomCommands(categories);
80 
81 
82  outStream.println("You may use the following commands in this console:\n");
83  for(Map.Entry<String, List<CommandHelpInfo>> entry : categories.entrySet()) {
84  outStream.println(entry.getKey() + ": ");
85  int maxCommandLength = 0;
86  for(CommandHelpInfo info : entry.getValue()) {
87  if (info.syntax.length() > maxCommandLength) {
88  maxCommandLength = info.syntax.length();
89  }
90  }
91  for(CommandHelpInfo info : entry.getValue()) {
92  outStream.println(pad(info.syntax, maxCommandLength + 3) + (info.abbreviation != null ? ("[" + info.abbreviation + "] ") : "") + info.shortDescription);
93  }
94  outStream.println();
95  }
96  }

106  private static void addAliases(LinkedHashMap<String, List<CommandHelpInfo>> categories) {
107  ArrayList<CommandHelpInfo> aliases = new ArrayList<>();
108  for(CommandAlias alias : InspectorConfiguration.getInstance().getAliases()) {
109  String aliasLeftPart = alias.getKey().trim();
110  String aliasRightPart = alias.getValue().trim();
111  for (int i = 0; i < alias.getNumberOfRequiredParameters(); i++) {
112  aliasLeftPart += " [arg" + i + "]";
113  }
114  for (int i = 0; i < alias.getNumberOfRequiredParameters(); i++) {
115  aliasRightPart = aliasRightPart.replace("{" + i + "}", "[arg" + i + "]");
116  }
117  if (alias.getNumberOfRequiredParameters() == 0 && alias.usesFullArgumentLine()) {
118  aliasLeftPart += " [argument]";
119  aliasRightPart = aliasRightPart.replace(CommandAlias.FULL_ARGUMENT_PATTERN, "[argument]");
120  }
121  aliases.add(new CommandHelpInfo(aliasLeftPart, null, aliasRightPart));
122  }
123  categories.put("Aliases", aliases);
124  }
