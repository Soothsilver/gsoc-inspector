146  private BreakpointStatus createBreakPointImpl(BreakpointCreationInformation newBP,
147  ExpressionBooleanInterface newBPExpression,
148  boolean hidden,
149  boolean firstHit) throws JPFInspectorGenericErrorException {
150  assert newBP != null;
151  assert newBPExpression != null;
152 
153  synchronized (breakpoints) {
154  InternalBreakpointHolder ibp = null;
155  if (newBP.getBPID() != BreakpointCreationInformation.BP_ID_NOT_DEFINED) {
156  ibp = breakpoints.get(newBP.getBPID());
157  if (ibp == null) {
158  throw new JPFInspectorGenericErrorException("Breakpoint with ID " + newBP.getBPID() + " does not exist and so can't be modified");
159  }
160  }
161  if (ibp == null) {
162  ibp = new InternalBreakpointHolder(newBP.getBPID(), callBacks, !hidden, firstHit);
163  }
164  ibp.modifyBPSettings(newBP, newBPExpression);
165  breakpoints.put(ibp.getBPID(), ibp);
166  // Check for error
167  if (newBP.getBPExpression() != null) {
168  if (!newBP.getBPExpression().trim().isEmpty()) {
169  return null;
170  }
171  }
172  return ibp.getBreakpointStatus(null);
173  } // end sync
174  }

176  private AssertStatus createAssertImpl (AssertCreationInformation newAssert, ExpressionBreakpointAssert newBPExpression) throws JPFInspectorGenericErrorException {
177  assert newAssert != null;
178  assert newBPExpression != null;
179 
180  synchronized (breakpoints) {
181  InternalAssertHolder iah = null;
182  if (newAssert.getBPID() != BreakpointCreationInformation.BP_ID_NOT_DEFINED) {
183  InternalBreakpointHolder ibp = breakpoints.get(newAssert.getBPID());
184  if (ibp == null) {
185  throw new JPFInspectorGenericErrorException("Assertion with ID " + newAssert.getBPID() + " does not exist so it can't be modified.");
186  }
187  if (ibp.isAssert() == false) {
188  throw new JPFInspectorGenericErrorException("The breakpoint with ID " + newAssert.getBPID() + " is not an assertion.");
189  }
190 
191  assert (ibp instanceof InternalAssertHolder);
192  iah = (InternalAssertHolder) ibp;
193  }
194  if (iah == null) {
195  iah = new InternalAssertHolder(newAssert.getBPID(), callBacks, true, false, newAssert.getPosition(), newAssert.getCondition());
196  }
197 
198  iah.modifyAssertSettings(newAssert, newBPExpression);
199 
200  breakpoints.put(iah.getBPID(), iah);
201 
202  // Check for error
203  if (newAssert.getBPExpression() != null) {
204  if (!newAssert.getBPExpression().trim().isEmpty()) {
205  return null;
206  }
207  }
208  return iah.getBreakpointStatus(null);
209  } // end sync
