64  @Override
65  public void execute(JPFInspectorClient client, JPFInspectorBackEndInterface inspector, final PrintStream outStream) {
66  if (type == CmdRunTypes.RUN) {
67  final ShellManager shellMgr = ShellManager.getManager();
68  // Check whether JPF is running or not (if not we start them)
69  final List<VerifyCommand> vcList = shellMgr.getCommands(VerifyCommand.class);
70  assert vcList != null;
71  assert !vcList.isEmpty(); // Verify command exists
72 
73  final VerifyCommand vc = vcList.get(0);
74  if (!vc.isVerifying()) {
75  // Executes the JPF in the separate thread
76  new Thread(() -> {
77  try {
78  ShellManager.getManager().fireCommand(vc);
79  } catch (Throwable t) {
80  outStream.println("Exception while starting/resuming JPF.");
81  outStream.println(t.getMessage());
82  t.printStackTrace(outStream);
83  }
84  }).start();
85  inspector.waitUntilJpfBecomesConnected();
86  } else {
87  // JPF is running
88  try {
89  inspector.start();
90  } catch (JPFInspectorException e) {
91  outStream.println(e.getMessage());
92  client.recordComment(e.getMessage());
93  }
94  }
95  } else if (type == CmdRunTypes.STOP) {
96  try {
97  inspector.stop();
98  } catch (JPFInspectorException e) {
99  outStream.println(e.getMessage());
100  client.recordComment(e.getMessage());
101  }
102  } else {
103  throw new RuntimeException("Unknown enum " + type.getClass().getName() + " entry " + type);
104  }
105  if (InspectorConfiguration.getInstance().shouldWaitAfterRun()) {
106  inspector.waitUntilStopped();
107  }
108  }
