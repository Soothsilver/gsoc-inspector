60  @Override
61  public void execute(JPFInspectorClient client, JPFInspectorBackEndInterface inspector, final PrintStream outStream) {
62  if (type == CmdRunTypes.RUN) {
63  final ShellManager shellMgr = ShellManager.getManager();
64  // Check whether JPF is running or not (if not we start them)
65  final List<VerifyCommand> vcList = shellMgr.getCommands(VerifyCommand.class);
66  assert vcList != null;
67  assert vcList.size() > 0; // Verify command exists
68 
69  final VerifyCommand vc = vcList.get(0);
70  if (!vc.isVerifying()) {
71  // Executes the JPF in the separate thread
72  new Thread(() -> {
73  try {
74  ShellManager.getManager().fireCommand(vc);
75  } catch (Throwable t) {
76  outStream.println("Exception while starting/resuming JPF.");
77  outStream.println(t.getMessage());
78  t.printStackTrace(outStream);
79  }
80  }).start();
81  inspector.waitUntilJpfBecomesConnected();
82  } else {
83  // JPF is running
84  try {
85  inspector.start();
86  } catch (JPFInspectorException e) {
87  outStream.println(e.getMessage());
88  client.recordComment(e.getMessage());
89  }
90  }
91  } else if (type == CmdRunTypes.STOP) {
92  try {
93  inspector.stop();
94  } catch (JPFInspectorException e) {
95  outStream.println(e.getMessage());
96  client.recordComment(e.getMessage());
97  }
98  } else {
99  throw new RuntimeException("Unknown enum " + type.getClass().getName() + " entry " + type);
100  }
101  }
