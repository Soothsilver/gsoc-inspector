121  @Override
122  public void backwardStep (StepType type) throws JPFInspectorGenericErrorException {
123  initialStopTest(true, "cannot execute backward step");
124  // According stepType find
125  // a) how much transitions it is required to backtrack
126  // b) create breakpoint for specific position in transition (handle repetitive invocations??)
127 
128  // Instantiate the creator and discover the instruction to backtrack to
129  InspectorState inspState = stopHolder.getInspectorState();
130  BackwardBreakpointCreator bbc;
131  if (type == StepType.ST_TRANSITION_DATA) {
132  throw new RuntimeException("Not yet implemented.");
133  } else if (type == StepType.ST_TRANSITION_SCHED) {
134  throw new RuntimeException("Not yet implemented.");
135  } else if (type == StepType.ST_TRANSITION_ALL) {
136  throw new RuntimeException("Not yet implemented.");
137  } else if (type == StepType.ST_INSTRUCTION) {
138  bbc = BackwardBreakpointCreator.getBackwardStepInstruction(inspState);
139  } else if (type == StepType.ST_LINE) {
140  bbc = BackwardBreakpointCreator.getBackwardStepLine(inspState);
141  } else if (type == StepType.ST_STEP_IN) {
142  bbc = BackwardBreakpointCreator.getBackwardStepIn(inspState);
143  } else if (type == StepType.ST_STEP_OUT) {
144  bbc = BackwardBreakpointCreator.getBackwardStepOut(inspState);
145  } else {
146  throw new RuntimeException("Unsupported " + type.getClass().getSimpleName() + " entry " + type);
147  }
148  if (bbc == null) {
149  throw new JPFInspectorGenericErrorException("Backwards step not possible.");
150  }
151 
152  // Create the breakpoint on that specific instruction
153  int bpID = bbc.createBreakpoint(breakpointMgr);
154 
155  // Enable silent mode in JPF Listener
156  InspectorListener listener = inspector.getInspectorListener();
157  assert listener != null : "Internal error - if JPF is connected then Listener has to be set";
158  InspectorListenerModeSilent listenerSilentMode =
159  new InspectorListenerModeSilent(inspector, this, breakpointMgr,
160  bbc.getTransitionsToBacktrack(),
161  bpID, dftMgr, stopHolder);
162  listener.pushMode(listenerSilentMode);
163 
164  Search search = inspState.getSearch();
165  assert search != null : "Internal error - not specified search";
166 
167  // reset the root CG if we might backtrack to it
168  ChoiceGenerator[] allCGs = search.getVM().getSystemState().getChoiceGenerators();
169  if (bbc.getTransitionsToBacktrack() >= allCGs.length) allCGs[0].reset();
170 
171  // Stop current transition (to prevent invoke more instruction than necessary) - only if this makes sense - instruction/throw/object_created
172  //search.getVM().breakTransition(); // We cannot add new transition
173  //search.requestBacktrack();
174  search.getVM().ignoreState();
175 
176  if (DEBUG) {
177  Debugging.getSwingShellLogger().warning("Backstep commences, backtrack " + bbc.getTransitionsToBacktrack() + " transitions to breakpoint \"" + bbc.getBreakpointHitCondition().getNormalizedExpression() + "\"");
178  }
179 
180  // Resume execution -> now silent backtrack and "breakpoint hit in single forward step should occur"
181  stopHolder.resumeExecution();
182 
183  /*
184  * I don't get what this means. It's probably the steps that happen, in order, in the silent mode:
185  *
186  * Rest is done in the listenerSilentMode
187  * Receives silent backtrack notifications
188  * Updates default trace
189  * Restores CG to same choice as before
190  * Start forward execution
191  * In the this.notifyBackwardStep()
192  * Enable normal mode in JPF Listener
193  */
194  }

196  @Override
197  public void forwardStep (StepType type) throws JPFInspectorGenericErrorException {
198  initialStopTest(true, "cannot execute forward step");
199 
200  BreakpointCreationExpression newBP = new BreakpointCreationExpression();
201  newBP.setBounds(null, null, "<=", 1);
202  newBP.setState(BreakpointState.BP_STATE_ENABLED);
203 
204  ExpressionBoolean bpExpression;
205  if (type == StepType.ST_TRANSITION_DATA) {
206  bpExpression = new ExpressionBreakpointChoiceGenerator(BreakPointModes.BP_MODE_CHOICE_DATA);
207  } else if (type == StepType.ST_TRANSITION_SCHED) {
208  bpExpression = new ExpressionBreakpointChoiceGenerator(BreakPointModes.BP_MODE_CHOICE_SCHEDULING);
209  } else if (type == StepType.ST_TRANSITION_ALL) {
210  bpExpression = new ExpressionBreakpointChoiceGenerator(BreakPointModes.BP_MODE_CHOICE_BOTH);
211  } else if (type == StepType.ST_INSTRUCTION) {
212  bpExpression = new ExpressionBreakpointInstructionType(InstructionType.IT_ANY);
213  } else if (type == StepType.ST_LINE) {
214  bpExpression = new ExpressionBreakpointSingleStep(inspector, stopHolder.getJVM(), LocationTypes.LT_POSITION_LEAVED_STEP_OVER);
215  } else if (type == StepType.ST_STEP_IN) {
216  bpExpression = new ExpressionBreakpointSingleStep(inspector, stopHolder.getJVM(), LocationTypes.LT_POSITION_LEAVED_STEP_IN);
217  } else if (type == StepType.ST_STEP_OUT) {
218  ThreadInfo ti = stopHolder.getJVM().getCurrentThread();
219  bpExpression = ExpressionBreakpointStepOut.getStepOutToCaller(ti);
220  } else {
221  throw new RuntimeException("Unsupported " + type.getClass().getSimpleName() + " entry " + type);
222  }
223 
224  assert (bpExpression != null);
225  breakpointMgr.createInternalBreakpoint(newBP, bpExpression, true);
226 
227  stopHolder.resumeExecution();
228  }

253  public boolean initialStopTest (boolean wait, String msg) throws JPFInspectorGenericErrorException {
254  if (inspector.getJPF() == null) {
255  throw new JPFInspectorGenericErrorException("No JPF instance to observe");
256  }
257  boolean wasStopped = stopHolder.isStopped();
258  if (!wasStopped) {
259  if (!wait) {
260  throw new JPFInspectorGenericErrorException("SuT is running - " + msg);
261  } else {
262  run = false;
263  stopHolder.waitUntilStopped();
264  }
265  }
266  return wasStopped;
267  }
