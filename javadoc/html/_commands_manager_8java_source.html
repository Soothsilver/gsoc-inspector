158  @Override
159  public void backstep(StepType type) throws JPFInspectorGenericErrorException {
160  stopHolder.waitUntilStopped();
161 
162  // Instantiate the creator and discover the instruction to backtrack to
163  InspectorState inspState = stopHolder.getInspectorState();
164  BackwardBreakpointCreator bbc;
165  if (type == StepType.BACK_BREAKPOINT_HIT) {
166  bbc = BackwardBreakpointCreator.getBackBreakpointHit(breakpointHandler.getLastBreakpointHitLocation(), inspState);
167  } else if (type == StepType.ST_INSTRUCTION) {
168  bbc = BackwardBreakpointCreator.getBackwardStepInstruction(inspState);
169  } else if (type == StepType.ST_LINE) {
170  bbc = BackwardBreakpointCreator.getBackwardStepLine(inspState);
171  } else if (type == StepType.ST_STEP_IN) {
172  bbc = BackwardBreakpointCreator.getBackwardStepIn(inspState);
173  } else if (type == StepType.ST_STEP_OUT) {
174  bbc = BackwardBreakpointCreator.getBackwardStepOut(inspState);
175  } else {
176  throw new RuntimeException("Unsupported " + type.getClass().getSimpleName() + " entry " + type);
177  }
178  if (bbc == null) {
179  throw new JPFInspectorGenericErrorException(
180  "Backwards step not possible (there is no appropriate step left for this thread to backtrack to).");
181  }
182 
183 
184  createBackwardsBreakpointAndResumeExecution(inspState, bbc);
185 
186  }

199  private void createBackwardsBreakpointAndResumeExecution(InspectorState inspState, BackwardBreakpointCreator bbc) throws JPFInspectorGenericErrorException {
200  assert bbc != null;
201 
202  // Create the breakpoint on that specific instruction
203  int bpID = bbc.createBreakpoint(breakpointHandler);
204  BreakpointStatus breakpointStatus = breakpointHandler.getBreakpoint(bpID);
205  Debugging.getLogger()
206  .info("The following breakpoint would be created:\n--> " + breakpointStatus.getNormalizedBreakpointExpression());
207  Debugging.getLogger().info("We will need to backtrack " + bbc.getTransitionsToBacktrack() + " transitions.");
208 
209 
210  // Enable silent mode in JPF Listener
211  InspectorListener listener = inspector.getInspectorListener();
212  assert listener != null : "Internal error - if JPF is connected then Listener has to be set";
213  InspectorListenerModeSilent listenerSilentMode =
214  new InspectorListenerModeSilent(inspector, this, breakpointHandler,
215  bbc.getTransitionsToBacktrack(),
216  bpID, dftMgr, stopHolder);
217  listener.pushMode(listenerSilentMode);
218 
219  Search search = inspState.getSearch();
220  assert search != null : "Internal error - not specified search";
221 
222  // reset the root CG if we might backtrack to it
223  /*
224  ChoiceGenerator<?>[] allCGs = search.getVM().getSystemState().getChoiceGenerators();
225  if (bbc.getTransitionsToBacktrack() >= allCGs.length) {
226  allCGs[0].reset();
227  }
228  */
229 
230  // Stop current transition (to prevent invoke more instruction than necessary) - only if this makes sense - instruction/throw/object_created
231  //search.getVM().breakTransition(); // We cannot add new transition
232  //search.requestBacktrack();
233  search.getVM().ignoreState();
234 
235  // Resume execution -> now silent backtrack and "breakpoint hit in single forward step should occur"
236  this.inspector.getChoiceGeneratorsManager().resumeWasDueToStepping();
237  stopHolder.resumeExecution();
238 
239  // The rest happens in the silent-mode Inspector listener.
240  // 1. It backtracks the determined number of transitions.
241  // 1a. During this, it updates the default forward trace.
242  // 2. It restores the first transition's CG to the same choice as before.
243  // 3. It starts forward execution until it reaches the breakpoint we just created.
244  // 4. It calls this.notifyBackwardStep().
245  // 4a. In which the normal notifications-mode Inspector listener is reactived and the backstepping ends.
246  }

269  @Override
270  public void forwardStep (StepType type) throws JPFInspectorGenericErrorException {
271  stopHolder.waitUntilStopped();
272 
273  BreakpointCreationExpression newBP = new BreakpointCreationExpression();
274  newBP.setBounds(null, null, "<=", 1);
275  newBP.setState(BreakpointState.ENABLED);
276 
277  ExpressionBoolean bpExpression;
278  if (type == StepType.ST_TRANSITION_DATA) {
279  bpExpression = new ExpressionBreakpointChoiceGenerator(BreakPointModes.BP_MODE_CHOICE_DATA);
280  } else if (type == StepType.ST_TRANSITION_SCHED) {
281  bpExpression = new ExpressionBreakpointChoiceGenerator(BreakPointModes.BP_MODE_CHOICE_SCHEDULING);
282  } else if (type == StepType.ST_TRANSITION_ALL) {
283  bpExpression = new ExpressionBreakpointChoiceGenerator(BreakPointModes.BP_MODE_CHOICE_BOTH);
284  } else if (type == StepType.ST_INSTRUCTION) {
285  bpExpression = new ExpressionBreakpointInstructionType(InstructionType.ANY);
286  } else if (type == StepType.ST_LINE) {
287  bpExpression = new ExpressionBreakpointSingleStep(inspector, stopHolder.getVM(), LocationTypes.LT_POSITION_LEAVED_STEP_OVER);
288  } else if (type == StepType.ST_STEP_IN) {
289  bpExpression = new ExpressionBreakpointSingleStep(inspector, stopHolder.getVM(), LocationTypes.LT_POSITION_LEAVED_STEP_IN);
290  } else if (type == StepType.ST_STEP_OUT) {
291  ThreadInfo ti = stopHolder.getVM().getCurrentThread();
292  bpExpression = ExpressionBreakpointStepOut.getStepOutToCaller(ti);
293  } else {
294  throw new RuntimeException("Unsupported " + type.getClass().getSimpleName() + " entry " + type);
295  }
296 
297  assert (bpExpression != null);
298  breakpointHandler.createInternalBreakpoint(newBP, bpExpression);
299 
300  stopHolder.resumeExecution();
301  }

316 
317  if (!success) {
318  serverCallbacks.genericError("Backward step failed (" + reason + ")");
