58  @Override
59  public boolean evaluateExpression (InspectorState state) {
60  assert state != null;
61  if (DEBUG) {
62  inspector.getDebugPrintStream().println(this.getClass().getSimpleName() + ".evaluateExpression(...)");
63  }
64  if (state.getListenerMethod() != ListenerMethod.LM_EXECUTE_INSTRUCTION) {
65  return false;
66  }
67 
68  VM vm = state.getVM();
69  assert vm != null;
70 
71  final Instruction thisInstruction = vm.getInstruction();
72 
73  // This represents whether we are the same file and line, but we still need to ensure that we are the FIRST
74  // instruction on this line. By "first instruction", we mean "the first instruction on this line in the current method".
75  if (!targetLocation.hitPosition(thisInstruction)) {
76  return false;
77  }
78 
79  int targetLine = targetLocation.getLineNumber();
80  Instruction firstInstructionOnThisLine = thisInstruction.getMethodInfo().getInstructionsForLine(targetLine)[0];
81  return thisInstruction.equals(firstInstructionOnThisLine);
82  }
