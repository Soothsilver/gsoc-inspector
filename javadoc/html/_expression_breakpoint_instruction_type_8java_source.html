50  @Override
51  public boolean evaluateExpression (InspectorState state) {
52  assert state != null;
53  if (state.getListenerMethod() != ListenerMethod.LM_INSTRUCTION_EXECUTED) {
54  return false;
55  }
56  VM vm = state.getJVM();
57  Instruction inst = MigrationUtilities.getLastInstruction(vm);
58 
59  if (inst == null) {
60  return false;
61  } else if (instType == InstructionType.IT_ANY) {
62  return true;
63  } else if (instType == InstructionType.IT_ARRAY) {
64  return (inst instanceof ArrayElementInstruction);
65  } else if (instType == InstructionType.IT_FIELD_ACCESS) {
66  return (inst instanceof FieldInstruction);
67  } else if (instType == InstructionType.IT_FIELD_READ) {
68  return (inst instanceof GETFIELD) || (inst instanceof GETSTATIC);
69  } else if (instType == InstructionType.IT_FIELD_WRITE) {
70  return (inst instanceof PUTFIELD) || (inst instanceof PUTSTATIC);
71  } else if (instType == InstructionType.IT_IFCOND) {
72  return (inst instanceof IfInstruction);
73  } else if (instType == InstructionType.IT_INVOKE) {
74  return (inst instanceof InvokeInstruction);
75  } else if (instType == InstructionType.IT_LOCK) {
76  return (inst instanceof LockInstruction);
77  } else if (instType == InstructionType.IT_RETURN) {
78  return (inst instanceof ReturnInstruction);
79  } else {
80  throw new RuntimeException("Internal error - Unknown enum " + InstructionType.class.getSimpleName() + " entry " + instType);
81  }
82  }
