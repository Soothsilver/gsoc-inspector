47  @Override
48  public boolean evaluateExpression (InspectorState state) {
49  assert state != null;
50  if (state.getListenerMethod() != ListenerMethod.LM_EXECUTE_INSTRUCTION) {
51  return false;
52  }
53  VM vm = state.getVM();
54  Instruction inst = vm.getInstruction();
55 
56  if (inst == null) {
57  return false;
58  } else if (instType == InstructionType.ANY) {
59  return true;
60  } else if (instType == InstructionType.ARRAY) {
61  return (inst instanceof ArrayElementInstruction);
62  } else if (instType == InstructionType.FIELD_ACCESS) {
63  return (inst instanceof FieldInstruction);
64  } else if (instType == InstructionType.FIELD_READ) {
65  return (inst instanceof GETFIELD) || (inst instanceof GETSTATIC);
66  } else if (instType == InstructionType.FIELD_WRITE) {
67  return (inst instanceof PUTFIELD) || (inst instanceof PUTSTATIC);
68  } else if (instType == InstructionType.IFCOND) {
69  return (inst instanceof IfInstruction);
70  } else if (instType == InstructionType.INVOKE) {
71  return (inst instanceof InvokeInstruction);
72  } else if (instType == InstructionType.LOCK) {
73  return (inst instanceof LockInstruction);
74  } else if (instType == InstructionType.RETURN) {
75  return (inst instanceof ReturnInstruction);
76  } else if (instType == InstructionType.LOCAL_ACCESS) {
77  return (inst instanceof gov.nasa.jpf.jvm.bytecode.JVMLocalVariableInstruction);
78  } else if (instType == InstructionType.LOCAL_READ) {
79  return (inst instanceof gov.nasa.jpf.jvm.bytecode.JVMLocalVariableInstruction) &&
80  !(inst instanceof StoreInstruction);
81  } else if (instType == InstructionType.LOCAL_WRITE) {
82  return (inst instanceof gov.nasa.jpf.jvm.bytecode.JVMLocalVariableInstruction) &&
83  (inst instanceof StoreInstruction);
84  } else {
85  throw new RuntimeException("Internal error - Unknown enum " + InstructionType.class.getSimpleName() + " entry " + instType);
86  }
87  }
