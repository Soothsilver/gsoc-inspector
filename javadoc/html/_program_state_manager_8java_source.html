62  @Override
63  public Map<Integer, PSEThread> getThreads (Integer threadNum) throws JPFInspectorException {
64  stopHolder.waitUntilStopped();
65 
66  Map<Integer, PSEThread> result;
67 
68  VM vm = getJVM();
69  if (vm == null) {
70  return null;
71  }
72 
73  if (threadNum != null) {
74  result = new HashMap<>(1);
75  StateThreadInfo sti = new StateThreadInfo(inspector, vm, threadNum);
76  result.put(threadNum, sti.toHierarchy3(this.attributesManager));
77 
78  } else {
79  ThreadList tl = vm.getThreadList();
80  result = new HashMap<>(tl.length());
81 
82  for (ThreadInfo ti : tl) {
83  StateThreadInfo sti = new StateThreadInfo(inspector, vm, ti.getId());
84  result.put(ti.getId(), sti.toHierarchy3(this.attributesManager));
85  }
86  }
87 
88  return result;
89  }

91  @Override
92  public Map<Integer, InstructionPosition> getThreadsPC (Integer threadNum) throws JPFInspectorGenericErrorException {
93  stopHolder.waitUntilStopped();
94 
95  VM vm = getJVM();
96  if (vm == null) {
97  return null;
98  }
99 
100  ThreadList tl = vm.getThreadList();
101 
102  Map<Integer, InstructionPosition> result;
103  if (threadNum != null) {
104  result = new HashMap<>(1);
105  } else {
106  result = new HashMap<>(tl.length());
107  }
108 
109  for (ThreadInfo ti : tl) {
110  if (!((threadNum == null) || (ti.getId() == threadNum))) {
111  continue;
112  }
113  Instruction pc = ti.getPC();
114  if (pc == null) {
115  // Terminated threads -> no position to store
116  continue;
117  }
118  result.put(ti.getId(), InstructionPositionImpl.getInstructionPosition(pc));
119  }
120  return result;
121  }

123  @Override
124  public ProgramStateEntry evaluateStateExpression (String expr) throws JPFInspectorException {
125  stopHolder.waitUntilStopped();
126 
127  // Create a parse tree (hierarchy 1)
128  ExpressionStateRootNode parsedExpr = parser.getExpressionStateInterface(expr);
129  if (parsedExpr == null) {
130  return null;
131  }
132 
133  // Create "representation of parse tree according current state" (hierarchy 2)
134  StateNodeInterface sni = parsedExpr.getResultExpression(inspector, stopHolder.getInspectorState());
135 
136  // Create client representation of the state (hiearchy 3)
137  return sni.toHierarchy3(this.attributesManager);
138  }
139 
140  private VM getJVM () throws JPFInspectorGenericErrorException {
141  VM vm = stopHolder.getVM();
142  if (vm == null) {
143  throw new JPFInspectorGenericErrorException("Internal error - JVM not as a part of the state (JPF is not connected");
144  }
145  return vm;
146  }

165  StateReadableValue rVal = esa.getRValue(inspector, stopHolder.getInspectorState());
