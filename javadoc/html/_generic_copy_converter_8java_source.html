33  public AttributeConversionResult stringToAttribute(String newAttributeValue) {
34  ExpressionParser expressionParser = new ExpressionParser(inspector.getServer());
35  try {
36  ExpressionStateRootNode root = expressionParser.getExpressionStateInterface(newAttributeValue);
37  StateNodeInterface resultExpression = root.getResultExpression(inspector.getServer(), inspector.getServer()
38  .getStopHolder()
39  .getInspectorState());
40  if (!(resultExpression instanceof StateWritableValue)) {
41  throw new JPFInspectorGenericErrorException("You referenced an object that cannot have attributes attached.");
42  }
43  StateWritableValue value = (StateWritableValue)resultExpression;
44  AttributeAttachment attributes = value.getAttributeAttachment();
45  Iterable<Object> iterable = attributes.getAttributes();
46  Object firstItem = null;
47  for (Object iterableItem : iterable) {
48  if (firstItem != null) {
49  throw new JPFInspectorGenericErrorException("The referenced object has multiple attributes. In JPF Inspector, a converter can only create a single attribute.");
50  }
51  firstItem = iterableItem;
52  }
53  return AttributeConversionResult.successful(firstItem);
54 
55  } catch (Exception exception) {
56  return AttributeConversionResult.failed("Attribute could not be copied: " + exception.getMessage());
57  }
