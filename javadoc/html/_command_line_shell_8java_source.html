50  Config config = ShellManager.getManager().getConfig();
51 
52  Scanner scanner = new Scanner(inputStream);
53  inspector = JPFInspectorFacade.getInspectorClient(config.getTarget(), outputStream);
54  ShellManager.getManager().addCommandListener(VerifyCommand.class, this);
55 
56  if (!batchMode) {
57  outputStream.println("This is the JPF Inspector console for debugging the target \"" + config.getTarget() + "\".");
58  outputStream.println("Type \"hello\" to test if the Inspector is working or \"help\" to get a list of commands.");
59  outputStream.print(Constants.PROMPT);
60  }
61  while (scanner.hasNextLine()) {
62  String command = scanner.nextLine().trim();
63 
64  if (!command.isEmpty()) {
65  if (batchModeEchoInput) {
66  outputStream.println(Constants.PROMPT + command);
67  }
68  inspector.executeCommand(command, ExecutionContext.FROM_COMMAND_LINE_TERMINAL);
69  }
70 
71  if (!batchMode) {
72  outputStream.print(Constants.PROMPT);
73  }
74  }
75  // No more commands will be given.
76  // Let's terminate JPF then.
77  if (inspector.isPaused()) {
78  inspector.executeCommand("wait", ExecutionContext.FROM_COMMAND_LINE_TERMINAL);
79  inspector.executeCommand("terminate", ExecutionContext.FROM_COMMAND_LINE_TERMINAL);
80  }
