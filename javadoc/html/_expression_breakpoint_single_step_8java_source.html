77  public static ExpressionBreakpointSingleStep createBreakpointSingleStep (JPFInspector inspector, StepType stepType) throws JPFInspectorGenericErrorException {
78  assert inspector != null;
79  StopHolder sh = inspector.getStopHolder();
80  assert sh != null;
81  InspectorState inspState = sh.getInspectorState();
82  if (inspState == null) {
83  throw new JPFInspectorGenericErrorException(
84  "cannot create breakpoint, JPF has to be connected and stopped - program state is required to create given breakpoint.");
85  }
86 
87  VM vm = inspState.getJVM();
88  assert (vm != null);
89 
90  LocationTypes lt = null;
91  assert (stepType != null);
92  if (stepType == StepType.ST_STEP_IN) {
93  lt = LocationTypes.LT_POSITION_LEAVED_STEP_IN;
94  } else if (stepType == StepType.ST_LINE) {
95  lt = LocationTypes.LT_POSITION_LEAVED_STEP_OVER;
96  } else {
97  throw new JPFInspectorGenericErrorException("Step type (" + stepType + ") is not supporte by " + ExpressionBreakpointSingleStep.class.getSimpleName()
98  + " breakpoint.");
99  }
100 
101  return new ExpressionBreakpointSingleStep(inspector, vm, lt);
102  }

109  public ExpressionBreakpointSingleStep (JPFInspector inspector, VM vm, LocationTypes posHandling) {
110  // InstructionPosition pos, int threadNum, Transition reqTransition,
111  assert vm != null;
112  assert inspector != null;
113 
114  this.inspector = inspector;
115  this.posHandling = posHandling;

146  @Override
147  public boolean evaluateExpression (InspectorState state) {
148  assert state != null;
149  if (DEBUG) {
150  inspector.getDebugPrintStream().println(this.getClass().getSimpleName() + ".evaluateExpression(...)");
151  }
152  if (state.getListenerMethod() != ListenerMethod.LM_INSTRUCTION_EXECUTED) {
153  return false;
154  }
155 
156  VM vm = state.getJVM();
157  assert vm != null;
158 
159  // Check if we are in the same thread
160  int lastThread = MigrationUtilities.getThreadNumber(vm);
161  if (lastThread != threadNum) {
162  return false;
163  }
164 
165  // Check if we don't backtrack to depth (before place where single step starts)
166  if (!checkPath(vm, reqTransition)) {
167  return breakIfNoNextLine;
168  }
169 
170  final Path path = vm.getPath();
171  final Instruction lastInstr = MigrationUtilities.getLastInstruction(vm);
172  // final Step lastStep = vm.getSystemState().getTrail().getLastStep();
173  // final Instruction xxx = lastStep.getInstruction();
174  // assert lastInstr.equals(xxx);
175 
176  final boolean lastInstrHitPos = instPos.hitPosition(lastInstr);
177  if (DEBUG) {
178  inspector.getDebugPrintStream().println(
179  "\tlasInstr=" + ExpressionBreakpointPosition.instructionPosition(lastInstr) + "\n\tlastInstrHitPos=" + lastInstrHitPos);
180  }
181 
182  Instruction prevInstr = ExpressionBreakpointPosition.getInstructionForThread(vm.getSystemState().getTrail(), path, lastThread, 1);
183  if (prevInstr instanceof ReturnInstruction) {
184  // Implies that previous instruction in method is invoke (no jump, part of single basic block)
185  prevInstr = lastInstr.getPrev();
186  assert (prevInstr instanceof InvokeInstruction);
187  }
188  if (DEBUG) {
189  inspector.getDebugPrintStream().println("\tprevInstr=" + ExpressionBreakpointPosition.instructionPosition(prevInstr));
190  }
191 
192  if (posHandling == LocationTypes.LT_POSITION_LEAVED_STEP_OVER) {
193  ThreadInfo currentThread = vm.getCurrentThread();
194  if (containsStackFrame(currentThread, topStackFrame)) {
195  // Stack frame of method which current(last) instruction belongs to
196  StackFrame stackFrame = currentThread.getTopFrame();
197  if (lastInstr instanceof InvokeInstruction) {
198  // top Stack frame is callee frame, we are interested in caller stack frame
199  stackFrame = stackFrame.getPrevious();
200  }
201  if (stackFrame.equals(topStackFrame)) {
202  if (lastInstr instanceof ReturnInstruction) {
203  // Return from call -> check next instruction
204  return false;
205  }
206  return !lastInstrHitPos;
207  } else {
208  return false;
209  }
210  } else {
211  return true;
212  }
213  } else if (posHandling == LocationTypes.LT_POSITION_LEAVED_STEP_IN) {
214  if (lastInstrHitPos == true) {
215  return false;
216  }
217  if (lastInstr instanceof InvokeInstruction) {
218  return true;
219  }
220 
221  final boolean prevInstrHitPos = instPos.hitPosition(prevInstr);
222  if (DEBUG) {
223  inspector.getDebugPrintStream().println("tprevInstrHitPos=" + prevInstrHitPos);
224  }
225  return prevInstrHitPos;
226  } else {
227  throw new RuntimeException("Internal error - Unknown enum " + (posHandling != null ? posHandling.getClass() : LocationTypes.class).getSimpleName()
228  + " entry " + posHandling);
229  }
230  }

253  private static boolean checkPath (VM jvm, Transition tr) {
254  assert jvm != null;
255  Transition curTr = jvm.getCurrentTransition();
256  if (curTr.equals(tr)) {
257  return true;
258  }
259 
260  Iterator<Transition> itTransitions = jvm.getPath().iterator();
261  while (itTransitions.hasNext()) {
262  Transition pathTr = itTransitions.next();
263  if (pathTr.equals(tr)) {
264  return true;
265  }
266  }
267  return false;
268  }

270  private static boolean containsStackFrame (ThreadInfo ti, StackFrame sf) {
271 
272  Iterator<StackFrame> sfIt = ti.iterator();
273  while (sfIt.hasNext()) {
274  StackFrame threadsSF = sfIt.next();
275  if (threadsSF.equals(sf)) {
276  return true;
277  }
278  }
279  return false;
280  }
