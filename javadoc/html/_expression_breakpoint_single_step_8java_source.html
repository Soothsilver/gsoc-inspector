101  public static ExpressionBreakpointSingleStep createBreakpointSingleStep (JPFInspector inspector, StepType stepType) throws JPFInspectorGenericErrorException {
102  assert inspector != null;
103  StopHolder sh = inspector.getStopHolder();
104  assert sh != null;
105  InspectorState inspState = sh.getInspectorState();
106  if (inspState == null) {
107  throw new JPFInspectorGenericErrorException(
108  "cannot create breakpoint, JPF has to be connected and stopped - program state is required to create given breakpoint.");
109  }
110 
111  VM vm = inspState.getVM();
112  assert (vm != null);
113 
114  LocationTypes lt;
115  assert (stepType != null);
116  if (stepType == StepType.ST_STEP_IN) {
117  lt = LocationTypes.LT_POSITION_LEAVED_STEP_IN;
118  } else if (stepType == StepType.ST_LINE) {
119  lt = LocationTypes.LT_POSITION_LEAVED_STEP_OVER;
120  } else {
121  throw new JPFInspectorGenericErrorException("Step type (" + stepType + ") is not supported by " + ExpressionBreakpointSingleStep.class.getSimpleName()
122  + " breakpoint.");
123  }
124 
125  return new ExpressionBreakpointSingleStep(inspector, vm, lt);
126  }

169  public boolean evaluateExpression (InspectorState state) {
170  assert state != null;
171  if (DEBUG) {
172  inspector.getDebugPrintStream().println(this.getClass().getSimpleName() + ".evaluateExpression(...)");
173  }
174  if (state.getListenerMethod() != ListenerMethod.LM_EXECUTE_INSTRUCTION) {
175  return false;
176  }
177 
178  VM vm = state.getVM();
179  assert vm != null;
180 
181  // Check if we are in the same thread
182  int currentThreadNumber = vm.getCurrentThread().getId();
183  if (currentThreadNumber != threadNum) {
184  return false;
185  }
186 
187  // Check if we don't backtrack to depth (before place where single step starts)
188  if (!checkPath(vm, reqTransition)) {
189  return breakIfNoNextLine;
190  }
191 
192  final Path path = vm.getPath();
193  final Instruction impendingInstruction = vm.getInstruction();
194  // final Step lastStep = vm.getSystemState().getTrail().getLastStep();
195  // final Instruction xxx = lastStep.getInstruction();
196  // assert lastInstr.equals(xxx);
197 
198  // Whether we are at the same line as we were when the stepping started.
199  final boolean lastInstrHitPos = instPos.hitPosition(impendingInstruction);
200 
201  if (DEBUG) {
202  inspector.getDebugPrintStream().println(
203  "\tlasInstr=" + ExpressionBreakpointPosition.instructionPosition(impendingInstruction) + "\n\tlastInstrHitPos=" + lastInstrHitPos);
204  }
205 
206  // The instruction that was just executed right now
207  Instruction prevInstr = state.getLastExecutedInstruction(currentThreadNumber);
208  // Instruction prevInstr = ExpressionBreakpointPosition.getInstructionForThread(vm.getSystemState().getTrail(), path, lastThread, 1);
209 
210  // TODO this may now be false:
211 
212  /*
213  TODO this probably does not make sense anymore
214  if (prevInstr instanceof ReturnInstruction) {
215  // Implies that previous instruction in method is invoke (no jump, part of single basic block)
216  prevInstr = impendingInstruction.getPrev();
217  assert (prevInstr instanceof InvokeInstruction);
218  }
219  */
220  if (DEBUG) {
221  inspector.getDebugPrintStream().println("\tprevInstr=" + ExpressionBreakpointPosition.instructionPosition(prevInstr));
222  }
223 
224  if (posHandling == LocationTypes.LT_POSITION_LEAVED_STEP_OVER) {
225  ThreadInfo currentThread = vm.getCurrentThread();
226  if (containsStackFrame(currentThread, topStackFrame)) {
227  // Stack frame of method which current(last) instruction belongs to
228  StackFrame stackFrame = currentThread.getTopFrame();
229  /*
230  see above
231  if (impendingInstruction instanceof InvokeInstruction) {
232  // top Stack frame is callee frame, we are interested in caller stack frame
233  stackFrame = stackFrame.getPrevious();
234  }
235  */
236  if (stackFrame.equals(topStackFrame)) {
237  // We are in the same method.
238  /*
239  see above
240  if (impendingInstruction instanceof ReturnInstruction) {
241  // Return from call -> check next instruction
242  return false;
243  }*/
244 
245  //noinspection RedundantIfStatement (I think it's clearer this way.)
246  if (lastInstrHitPos) {
247  return false; // We are still on the same line.
248  } else {
249  return true; // We are no longer on the same line.
250  }
251  } else {
252  // We have gone deeper into a call and need to wait until we exit.
253  return false;
254  }
255  } else {
256  // We have left the method we were in when the stepping was initiated.
257  // That is one of the trigger conditions of step_over.
258  return true;
259  }
260  } else if (posHandling == LocationTypes.LT_POSITION_LEAVED_STEP_IN) {
261  if (lastInstrHitPos == true) {
262  return false;
263  }
264  if (impendingInstruction instanceof InvokeInstruction) {
265  return true;
266  }
267 
268  final boolean prevInstrHitPos = instPos.hitPosition(prevInstr);
269  if (DEBUG) {
270  inspector.getDebugPrintStream().println("tprevInstrHitPos=" + prevInstrHitPos);
271  }
272  return prevInstrHitPos;
273  } else {
274  throw new RuntimeException("Internal error - Unknown enum " + (posHandling != null ? posHandling.getClass() : LocationTypes.class).getSimpleName()
275  + " entry " + posHandling);
276  }
