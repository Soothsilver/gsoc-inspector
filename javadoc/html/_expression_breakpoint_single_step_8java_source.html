90  @Override
91  public boolean evaluateExpression (InspectorState state) {
92  assert state != null;
93  if (DEBUG) {
94  inspector.getDebugPrintStream().println(this.getClass().getSimpleName() + ".evaluateExpression(...)");
95  }
96  if (state.getListenerMethod() != ListenerMethod.LM_EXECUTE_INSTRUCTION) {
97  return false;
98  }
99 
100  VM vm = state.getVM();
101  assert vm != null;
102 
103  // Check if we are in the same thread
104  int currentThreadNumber = vm.getCurrentThread().getId();
105  if (currentThreadNumber != threadNum) {
106  return false;
107  }
108 
109  // Check if we didn't backtrack too deep (before place where single step starts)
110  if (!checkPath(vm, initialTransition)) {
111  if (DEBUG) {
112  inspector.getDebugPrintStream().println("We backtracked using forward stepping!");
113  }
114  return true;
115  }
116 
117  final Instruction impendingInstruction = vm.getInstruction();
118 
119  // Whether we are at the same line as we were when the stepping started.
120  final boolean weAreStillOnTheSameLine = initialPosition.hitPosition(impendingInstruction);
121 
122  // The instruction that was last executed by this thread, if any.
123  Instruction prevInstr = state.getLastExecutedInstruction(currentThreadNumber);
124 
125  if (steppingKind == LocationTypes.LT_POSITION_LEAVED_STEP_OVER) {
126  ThreadInfo currentThread = vm.getCurrentThread();
127  if (currentThread.getStackDepth() >= initialStackDepth) {
128  // We haven't returned from the method yet.
129 
130  if (currentThread.getStackDepth() == initialStackDepth) {
131  // We are in the same method.
132 
133  //noinspection RedundantIfStatement (I think it's clearer this way.)
134  if (weAreStillOnTheSameLine) {
135  return false; // We are still on the same line.
136  } else {
137  return true; // We are no longer on the same line.
138  }
139  } else {
140  // We have gone deeper into a call and need to wait until we exit.
141  return false;
142  }
143  } else {
144  // We have left the method we were in when the stepping was initiated.
145  // That is one of the trigger conditions of step_over.
146  if (DEBUG) {
147  inspector.getDebugPrintStream().println("We stepped out of the method!");
148  }
149  return true;
150  }
151  } else if (steppingKind == LocationTypes.LT_POSITION_LEAVED_STEP_IN) {
152  if (weAreStillOnTheSameLine == true) {
153  return false;
154  }
155  if (impendingInstruction instanceof InvokeInstruction) {
156  return true;
157  }
158 
159  final boolean prevInstrHitPos = initialPosition.hitPosition(prevInstr);
160  if (DEBUG) {
161  inspector.getDebugPrintStream().println("tprevInstrHitPos=" + prevInstrHitPos);
162  }
163  return prevInstrHitPos;
164  } else {
165  throw new RuntimeException("Internal error - Unknown enum " + (steppingKind != null ? steppingKind.getClass() : LocationTypes.class).getSimpleName()
166  + " entry " + steppingKind);
167  }
168  }

188  private static boolean checkPath (VM jvm, Transition tr) {
189  assert jvm != null;
190  Transition curTr = jvm.getCurrentTransition();
191  if (curTr.equals(tr)) {
192  return true;
193  }
194 
195  for (Transition pathTr : jvm.getPath()) {
196  if (pathTr.equals(tr)) {
197  return true;
198  }
199  }
200  return false;
201  }
