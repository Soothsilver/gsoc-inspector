59  // String to attribute
60  @Override
61  public void setAttributeValue(String expression) throws JPFInspectorException {
62  if (stringToAttributeConverters.isEmpty()) {
63  throw new JPFInspectorGenericErrorException("No attribute-to-string converters are loaded.");
64  }
65  inspector.waitUntilStopped();
66  ExpressionStateAttributeAssignment parsedExpr = parser.getAttributeAssignment(expression);
67  StateWritableValue targetObject = parsedExpr.getTargetObject(inspector,
68  inspector.getStopHolder().getInspectorState());
69  AttributeAttachment attachment = targetObject.getAttributeAttachment();
70  String errorMessages = "";
71  for (StringToAttributeConverter converter : stringToAttributeConverters) {
72  AttributeConversionResult acr = attachment.convertToAttributeUsing(parsedExpr.getNewAttributeValue().trim(),
73  converter);
74  if (acr.isSuccess()) {
75  attachment.setAttribute(acr.getCreatedAttribute());
76  return;
77  } else {
78  if (acr.getReasonForRefusal() != null) {
79  errorMessages += converter.getClass().getSimpleName() + ": " + acr.getReasonForRefusal() + "\n";
80  }
81  }
82  }
83 
84  // We failed.
85  throw new JPFInspectorGenericErrorException("No converter was able to parse the new attribute value." +
86  (errorMessages.isEmpty() ? "" : "\n" + errorMessages));
87  }

101  private String getAttachmentAttributes(AttributeAttachment attributeAttachment) {
102  ArrayList<String> attributeStrings = new ArrayList<>();
