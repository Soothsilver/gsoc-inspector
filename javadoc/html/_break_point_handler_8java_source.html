144  private BreakPointStatus createBreakPointImpl(BreakPointCreationInformation newBP,
145  ExpressionBooleanInterface newBPExpression,
146  boolean hidden,
147  boolean firstHit) throws JPFInspectorGenericErrorException {
148  assert newBP != null;
149  assert newBPExpression != null;
150 
151  synchronized (breakpoints) {
152  InternalBreakpointHolder ibp = null;
153  if (newBP.getBPID() != BreakPointCreationInformation.BP_ID_NOT_DEFINED) {
154  ibp = breakpoints.get(newBP.getBPID());
155  if (ibp == null) {
156  throw new JPFInspectorGenericErrorException("Breakpoint with ID=" + newBP.getBPID() + " not exists --> cann't be modified");
157  }
158  }
159  if (ibp == null) {
160  ibp = new InternalBreakpointHolder(newBP.getBPID(), callBacks, !hidden, firstHit);
161  }
162  ibp.modifyBPSettings(newBP, newBPExpression);
163  breakpoints.put(ibp.getBPID(), ibp);
164  // Check for error
165  if (newBP.getBPExpression() != null) {
166  if (!newBP.getBPExpression().trim().isEmpty()) {
167  return null;
168  }
169  }
170  return ibp.getBreakpointStatus(null);
171  } // end sync
172  }

174  private AssertStatus createAssertImpl (AssertCreate newAssert, ExpressionBreakpointAssert newBPExpression) throws JPFInspectorGenericErrorException {
175  assert newAssert != null;
176  assert newBPExpression != null;
177 
178  synchronized (breakpoints) {
179  InternalAssertHolder iah = null;
180  if (newAssert.getBPID() != BreakPointCreationInformation.BP_ID_NOT_DEFINED) {
181  InternalBreakpointHolder ibp = breakpoints.get(newAssert.getBPID());
182  if (ibp == null) {
183  throw new JPFInspectorGenericErrorException("Assertion with ID " + newAssert.getBPID() + " does not exist so it can't be modified.");
184  }
185  if (ibp.isAssert() == false) {
186  throw new JPFInspectorGenericErrorException("The breakpoint with ID " + newAssert.getBPID() + " is not an assertion.");
187  }
188 
189  assert (ibp instanceof InternalAssertHolder);
190  iah = (InternalAssertHolder) ibp;
191  }
192  if (iah == null) {
193  iah = new InternalAssertHolder(newAssert.getBPID(), callBacks, true, false, newAssert.getPosition(), newAssert.getCondition());
194  }
195 
196  iah.modifyBPSettings(newAssert, newBPExpression);
197 
198  breakpoints.put(iah.getBPID(), iah);
199 
200  // Check for error
201  if (newAssert.getBPExpression() != null) {
202  if (!newAssert.getBPExpression().trim().isEmpty()) {
203  return null;
204  }
205  }
206  return iah.getBreakpointStatus(null);
207  } // end sync
208  }
